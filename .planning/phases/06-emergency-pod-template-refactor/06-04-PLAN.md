const emergencyOnstartB2 = `#!/bin/bash
set -euo pipefail
WEIGHTS_PATH=/weights/qwen/model.gguf
mkdir -p "$(dirname "$WEIGHTS_PATH")" /app/templates
if [[ ! -f "$WEIGHTS_PATH" ]]; then
  apt-get update && apt-get install -y curl ca-certificates >/dev/null
  curl -sL https://dl.min.io/client/mc/release/linux-amd64/mc -o /usr/local/bin/mc
  chmod +x /usr/local/bin/mc
  mc alias set ifix "$MINIO_ENDPOINT" "$MINIO_ACCESS_KEY" "$MINIO_SECRET_KEY" >/dev/null
  mc cp "ifix/${MINIO_BUCKET}/${WEIGHTS_QWEN_KEY}" "$WEIGHTS_PATH"
  ACTUAL=$(sha256sum "$WEIGHTS_PATH" | awk '{print $1}')
  [[ "$ACTUAL" = "$WEIGHTS_QWEN_SHA256" ]] || { echo "FATAL: Qwen sha256 mismatch (expected=$WEIGHTS_QWEN_SHA256 actual=$ACTUAL)" >&2; exit 1; }
fi
if [[ -n "${JINJA_TEMPLATE_KEY:-}" ]]; then
  JINJA_PATH=/app/templates/qwen3.5-27b-tool-calling.jinja
  if [[ ! -f "$JINJA_PATH" ]]; then
    mc cp "ifix/${MINIO_BUCKET}/${JINJA_TEMPLATE_KEY}" "$JINJA_PATH"
    JACT=$(sha256sum "$JINJA_PATH" | awk '{print $1}')
    [[ "$JACT" = "$JINJA_TEMPLATE_SHA256" ]] || { echo "FATAL: Jinja sha256 mismatch (expected=$JINJA_TEMPLATE_SHA256 actual=$JACT)" >&2; exit 1; }

fi
