  ```go
  // Package redisx (shed.go): Redis helpers for the Phase 5 shedding subsystem.
  //
  // Keys layout (namespace "gw:shed:"):
  //   - gw:shed:{upstream}       — Hash with {state, since_unix, reason, signals}
  //   - gw:shed:events           — Pub/Sub channel for state transitions
  //   - gw:shed:force:{upstream} — Shadow state for operator overrides (TTL-bounded)
  //
  // Semantics: in-process FSM is authoritative; Redis is a mirror.
  // Redis-down does NOT stop shedding from operating (CONTEXT D-C3).
  // Pub/Sub is at-most-once; periodic reconcile (shed/reconcile.go) closes
  // the convergence gap (RESEARCH Pitfall 3).
  package redisx

  import (
      "context"
      "encoding/json"
      "fmt"
      "time"

      "github.com/redis/go-redis/v9"
  )

  const (
      // ShedEventsChannel is the Pub/Sub channel name used for FSM transitions.
      ShedEventsChannel = "gw:shed:events"

      // shedStateKeyPrefix + {upstream} = Hash per upstream current state.
      shedStateKeyPrefix = "gw:shed:"

      // shedForceKeyPrefix + {upstream} = operator override shadow key (TTL).
      shedForceKeyPrefix = "gw:shed:force:"

      // redisOpTimeout caps individual Redis calls.
      redisOpTimeout = 2 * time.Second
  )

  // ShedEvent is the JSON payload published on ShedEventsChannel when a
  // FSM transitions. Consumers unmarshal and call Set.ApplyRemoteEvent.
  type ShedEvent struct {
      Upstream  string `json:"upstream"`
      State     string `json:"state"`      // "off" | "armed" | "on" | "recovering"
      SinceUnix int64  `json:"since_unix"` // time.Now().Unix() at transition
      Reason    string `json:"reason,omitempty"`
      // Signals may be nil if the publisher could not capture them.
      Signals *ShedEventSignals `json:"signals,omitempty"`
  }

  type ShedEventSignals struct {
      Inflight  int64  `json:"inflight,omitempty"`
      P95Ms     uint32 `json:"p95_ms,omitempty"`
      VramMiB   int64  `json:"vram_mib,omitempty"`
  }

  // ShedStateKey returns "gw:shed:{upstream}".
  func ShedStateKey(upstream string) string { return shedStateKeyPrefix + upstream }

  // ShedForceKey returns "gw:shed:force:{upstream}".
  func ShedForceKey(upstream string) string { return shedForceKeyPrefix + upstream }

  // WriteShedState mirrors the current FSM state to Redis as a Hash.
  // Best-effort — timeout 2s. Callers log failures via GatewayShedMirrorFailures.
  func WriteShedState(ctx context.Context, rdb *redis.Client, upstream, state, reason string, sinceUnix int64, sig *ShedEventSignals) error {
      if rdb == nil {
          return fmt.Errorf("redisx: nil client")
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      fields := map[string]any{
          "state":      state,
          "since_unix": sinceUnix,
          "reason":     reason,
      }
      if sig != nil {
          fields["inflight"] = sig.Inflight
          fields["p95_ms"] = sig.P95Ms
          fields["vram_mib"] = sig.VramMiB
      }
      return rdb.HSet(ctx, ShedStateKey(upstream), fields).Err()
  }

  // ReadShedState reads the Hash back. Returns (map, nil) on success; (nil, nil)
  // if key does not exist (new upstream, nothing published yet).
  func ReadShedState(ctx context.Context, rdb *redis.Client, upstream string) (map[string]string, error) {
      if rdb == nil {
          return nil, fmt.Errorf("redisx: nil client")
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      m, err := rdb.HGetAll(ctx, ShedStateKey(upstream)).Result()
      if err != nil {
          return nil, err
      }
      if len(m) == 0 {
          return nil, nil
      }
      return m, nil
  }

  // PublishShedEvent broadcasts a FSM transition to all replicas.
  func PublishShedEvent(ctx context.Context, rdb *redis.Client, ev ShedEvent) error {
      if rdb == nil {
          return fmt.Errorf("redisx: nil client")
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      payload, err := json.Marshal(ev)
      if err != nil {
          return err
      }
      return rdb.Publish(ctx, ShedEventsChannel, payload).Err()
  }

  // SubscribeShedEvents returns a PubSub attached to ShedEventsChannel.
  // The caller owns the PubSub (must Close on shutdown/reconnect).
  func SubscribeShedEvents(ctx context.Context, rdb *redis.Client) *redis.PubSub {
      return rdb.Subscribe(ctx, ShedEventsChannel)
  }

  // WriteShedForce sets the operator override with TTL. state must be "off" or "on".
  func WriteShedForce(ctx context.Context, rdb *redis.Client, upstream, state string, ttl time.Duration) error {
      if rdb == nil {
          return fmt.Errorf("redisx: nil client")
      }
      if state != "off" && state != "on" {
          return fmt.Errorf("redisx: invalid shed-force state %q (want off|on)", state)
      }
      if ttl <= 0 || ttl > time.Hour {
          return fmt.Errorf("redisx: shed-force TTL %s out of range (0 < ttl <= 1h)", ttl)
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      return rdb.Set(ctx, ShedForceKey(upstream), state, ttl).Err()
  }

  // GetShedForce returns (state, ttl, true) if override is active, or
  // ("", 0, false) if no active override exists.
  func GetShedForce(ctx context.Context, rdb *redis.Client, upstream string) (string, time.Duration, bool) {
      if rdb == nil {
          return "", 0, false
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      state, err := rdb.Get(ctx, ShedForceKey(upstream)).Result()
      if err != nil {
          return "", 0, false
      }
      ttl, err := rdb.PTTL(ctx, ShedForceKey(upstream)).Result()
      if err != nil || ttl < 0 {
          ttl = 0
      }
      return state, ttl, true
  }

  // DeleteShedForce clears an active override (used by `gatewayctl shed-force clear`).
  func DeleteShedForce(ctx context.Context, rdb *redis.Client, upstream string) error {
      if rdb == nil {
          return fmt.Errorf("redisx: nil client")
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      return rdb.Del(ctx, ShedForceKey(upstream)).Err()
  }

  // AllShedStateKeys returns all "gw:shed:{upstream}" keys currently present
  // (used by reconcile loop). Excludes "gw:shed:force:*" via MATCH pattern.
  func AllShedStateKeys(ctx context.Context, rdb *redis.Client) ([]string, error) {
      if rdb == nil {
          return nil, fmt.Errorf("redisx: nil client")
      }
      ctx, cancel := context.WithTimeout(ctx, redisOpTimeout)
      defer cancel()
      var keys []string
      var cursor uint64
      for {
          batch, next, err := rdb.Scan(ctx, cursor, "gw:shed:*", 100).Result()
          if err != nil {
              return nil, err
          }
          for _, k := range batch {
              // Filter out force keys (namespace collision by prefix).
              if len(k) > len(shedForceKeyPrefix) && k[:len(shedForceKeyPrefix)] == shedForceKeyPrefix {
                  continue
              }
              keys = append(keys, k)
          }
          cursor = next
          if cursor == 0 {
              break
          }
      }
      return keys, nil
  }
  ```

  ```go
  // Package shed (mirror.go): publishing helper for FSM transitions to Redis.
  // Follows the pattern of breaker/breaker.go publishTransition — best-effort,
  // fire-and-forget, silent fallback with metric counter on Redis failure.
  package shed

  // makePublishTransition returns a publisher bound to rdb. Called once at
  // Set construction (main.go wiring).

          return func(upstream string, to State, reason string, sig *redisx.ShedEventSignals) {}
      }
      return func(upstream string, to State, reason string, sig *redisx.ShedEventSignals) {
          ctx := context.Background()

              Reason:    reason,
              Signals:   sig,
          }
          if err := redisx.PublishShedEvent(ctx, rdb, ev); err != nil {
              obs.GatewayShedMirrorFailures.Inc()
          }
      }
  }

  func MakePublishTransition(rdb *redis.Client) publishTransitionFunc {
  ```

  ```go
  //
  // The subscribe goroutine reads gw:shed:events and calls Set.ApplyRemoteEvent
  // so other replicas' FSM transitions are visible on this replica's
  // remoteState map (consumed by dashboard + reconcile loop).
  //
  // This does NOT force the in-process FSM to the remote state — convergence
  // is handled by the periodic reconcile loop (reconcile.go) per RESEARCH Pitfall 3.
  package shed

  import (
      "context"
      "encoding/json"
      "log/slog"
      "time"

      "github.com/redis/go-redis/v9"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/redisx"
  )

  // Subscribe blocks reading from the shed events channel. On Redis
  // disconnects, the loop reconnects after 1s backoff.
  // Intended to be called via `go set.Subscribe(ctx)` from main.go wiring
  // AFTER HydrateFromRedis completes (Pitfall 3 boot rehydration).
  func (s *Set) Subscribe(ctx context.Context, rdb *redis.Client) {
      if rdb == nil {
          s.log.Warn("shed subscribe disabled — no Redis client")
          return
      }
      log := s.log.With("subsystem", "subscribe")
      for {
          if err := ctx.Err(); err != nil {
              return
          }
          ps := redisx.SubscribeShedEvents(ctx, rdb)
          ch := ps.Channel()
          drained := false
          for !drained {
              select {
              case <-ctx.Done():
                  _ = ps.Close()
                  return
              case msg, ok := <-ch:
                  if !ok {
                      drained = true
                      break
                  }
                  var ev redisx.ShedEvent
                  if err := json.Unmarshal([]byte(msg.Payload), &ev); err != nil {
                      log.Warn("malformed shed event", "payload", msg.Payload, "err", err)
                      continue
                  }
                  state := parseState(ev.State)
                  s.ApplyRemoteEvent(ev.Upstream, state)
                  log.Debug("applied remote shed event",
                      "upstream", ev.Upstream, "state", ev.State, "reason", ev.Reason)
              }
          }
          _ = ps.Close()
          log.Warn("shed pubsub channel closed; reconnecting")
          select {
          case <-ctx.Done():
              return
          case <-time.After(1 * time.Second):
          }
      }
  }

  func parseState(s string) State {
      switch s {
      case "off":
          return StateOff
      case "armed":
          return StateArmed
      case "on":
          return StateOn
      case "recovering":
          return StateRecovering
      }
      return StateOff
  }

  // HydrateFromRedis reads gw:shed:{upstream} for every currently-managed
  // upstream and populates remoteState. MUST be called ONCE at boot, BEFORE
  // Subscribe starts — otherwise the first few events may arrive before
  // the replica knows any remote state (RESEARCH Pitfall 3 mitigation).
  //
  // Does NOT alter in-process FSMs — remote state is advisory only.
  func (s *Set) HydrateFromRedis(ctx context.Context, rdb *redis.Client, log *slog.Logger) {
      if rdb == nil {
          return
      }
      for _, name := range s.Names() {
          m, err := redisx.ReadShedState(ctx, rdb, name)
          if err != nil {
              log.Warn("hydrate: read failed", "upstream", name, "err", err)
              continue
          }
          if m == nil {
              continue
          }
          state := parseState(m["state"])
          s.ApplyRemoteEvent(name, state)
          log.Debug("hydrate: applied remote state", "upstream", name, "state", state.String())
      }
  }
  ```

  ```go
  // Package shed (reconcile.go): periodic HGETALL loop that closes the
  // convergence gap left by lossy Pub/Sub (RESEARCH Pitfall 3).
  //
  // Every reconcileInterval (30s default), reads gw:shed:{upstream} Hash
  // for every managed upstream and compares with remoteState.
  // Divergence increments gateway_shed_mirror_reconcile_total{result="diverged"}
  // and updates remoteState to the Redis view.
  //
  // This is forward-compat for Phase 6 multi-replica — in Phase 5 single-
  // replica it is a sanity net.
  package shed

  import (
      "context"
      "log/slog"
      "time"

      "github.com/redis/go-redis/v9"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
      "github.com/ifixtelecom/gpu-ifix/gateway/internal/redisx"
  )

  // DefaultReconcileInterval is the default HGETALL cadence.
  const DefaultReconcileInterval = 30 * time.Second

  // ReconcileLoop blocks until ctx cancel, running one reconcile per interval.
  // Intended to be called via `go set.ReconcileLoop(ctx, rdb, interval, log)`
  // alongside Subscribe and the FSM ticker.
  func (s *Set) ReconcileLoop(ctx context.Context, rdb *redis.Client, interval time.Duration, log *slog.Logger) {
      if rdb == nil {
          return
      }
      if interval <= 0 {
          interval = DefaultReconcileInterval
      }
      log = log.With("module", "SHED_RECONCILE")
      t := time.NewTicker(interval)
      defer t.Stop()
      for {
          select {
          case <-ctx.Done():
              log.Info("reconcile loop stopping")
              return
          case <-t.C:
              s.reconcileOnce(ctx, rdb, log)
          }
      }
  }

  func (s *Set) reconcileOnce(ctx context.Context, rdb *redis.Client, log *slog.Logger) {
      diverged := 0
      ok := 0
      for _, name := range s.Names() {
          m, err := redisx.ReadShedState(ctx, rdb, name)
          if err != nil {
              obs.GatewayShedMirrorReconcile.WithLabelValues("error").Inc()
              continue
          }
          if m == nil {
              // No record in Redis yet — treat as ok (first boot).
              obs.GatewayShedMirrorReconcile.WithLabelValues("ok").Inc()
              ok++
              continue
          }
          redisState := parseState(m["state"])
          currentRemote, hasRemote := s.RemoteState(name)
          if !hasRemote || currentRemote != redisState {
              s.ApplyRemoteEvent(name, redisState)
              obs.GatewayShedMirrorReconcile.WithLabelValues("diverged").Inc()
              diverged++
              log.Debug("reconcile: remote state corrected",
                  "upstream", name, "from", currentRemote.String(), "to", redisState.String())
              continue
          }
          obs.GatewayShedMirrorReconcile.WithLabelValues("ok").Inc()
          ok++
      }
      if diverged > 0 {
          log.Info("reconcile completed with divergence", "diverged", diverged, "ok", ok)
      }
  }
  ```

  ```go
  // Package shed (tick.go): global FSM ticker goroutine.
  //
  // A single goroutine iterates all FSMs at 1Hz (configurable via
  // SHED_TICK_INTERVAL_MS), derives 2-of-3 Signals from the inflight
  // registry + latency rings + DCGM scraper, and calls FSM.Evaluate.
  //
  // Additionally, if an operator override is active in Redis
  // (gw:shed:force:{upstream}), the ticker forces the FSM to the override
  // state BEFORE Evaluate runs (D-C5). Override has TTL and expires
  // silently; metric gateway_shed_force_active reflects status.
  package shed

  import (
      "context"
      "log/slog"
      "time"

      "github.com/google/uuid"
      "github.com/redis/go-redis/v9"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
      "github.com/ifixtelecom/gpu-ifix/gateway/internal/redisx"
  )

  // VramReader abstracts the DCGM scraper so tests can provide mocks.
  type VramReader interface {
      ReadMiB() (int64, bool) // (value MiB, unknown)
  }

  // Thresholds per upstream, derived from upstreams.CircuitConfig at tick time.
  type Thresholds struct {
      InflightMax int64
      P95Ms       uint32
      VramMiB     int64
  }

  // TickerDeps is the bundle of state the RunTicker goroutine consumes.
  type TickerDeps struct {
      Set          *Set
      Inflight     *InflightRegistry
      Latency      map[string]*LatencyRing
      VramReader   VramReader                           // may be nil if DCGM disabled
      ThresholdSrc func(upstream string) Thresholds     // from upstreams.Loader.CircuitConfig
      Rdb          *redis.Client                        // optional; nil disables shed-force override
      Interval     time.Duration                        // default 1s
      TenantLabel  func(tenant uuid.UUID) string        // optional, for obs.GatewayInflightTenant labels (pass nil to skip)
  }

  // RunTicker blocks until ctx cancellation. Must be started via goroutine.
  func RunTicker(ctx context.Context, d TickerDeps, log *slog.Logger) {
      if d.Set == nil {
          return
      }
      interval := d.Interval
      if interval <= 0 {
          interval = 1 * time.Second
      }
      log = log.With("module", "SHED_FSM")
      t := time.NewTicker(interval)
      defer t.Stop()
      log.Info("shed FSM ticker started", "interval", interval)
      for {
          select {
          case <-ctx.Done():
              log.Info("FSM ticker stopping")
              return
          case now := <-t.C:
              d.runOneTick(ctx, now, log)
          }
      }
  }

  func (d *TickerDeps) runOneTick(ctx context.Context, now time.Time, log *slog.Logger) {
      d.Set.ForEach(func(upstream string, fsm *FSM) {
          // Apply shed-force override first, if active.
          if d.Rdb != nil {
              state, _, ok := redisx.GetShedForce(ctx, d.Rdb, upstream)
              if ok {
                  obs.GatewayShedForceActive.WithLabelValues(upstream).Set(1)
                  var target State
                  switch state {
                  case "off":
                      target = StateOff
                  case "on":
                      target = StateOn
                  }
                  if fsm.State() != target {
                      fsm.Transition(target, "operator_override")
                  }
                  return
              }
              obs.GatewayShedForceActive.WithLabelValues(upstream).Set(0)
          }

          th := d.ThresholdSrc(upstream)
          // If thresholds are zero-valued (shed feature disabled for this upstream,
          // e.g., tier-1), skip evaluation — FSM stays in its current state.
          if th.InflightMax == 0 && th.P95Ms == 0 && th.VramMiB == 0 {
              return
          }

          globalInflight := int64(0)
          if d.Inflight != nil {
              globalInflight = d.Inflight.GlobalInflight(upstream)
          }
          p95 := uint32(0)
          if ring, ok := d.Latency[upstream]; ok {
              p95 = ring.P95()
              obs.GatewayP95RequestMs.WithLabelValues(upstream).Set(float64(p95))
          }
          vramMiB, vramUnknown := int64(0), true
          if d.VramReader != nil {
              vramMiB, vramUnknown = d.VramReader.ReadMiB()
          }

          sig := Signals{
              InflightOverMax: th.InflightMax > 0 && globalInflight >= th.InflightMax,
              P95OverMax:      th.P95Ms > 0 && p95 >= th.P95Ms,
              VramOverMax:     th.VramMiB > 0 && vramMiB >= th.VramMiB,
              VramUnknown:     vramUnknown,
          }
          fsm.Evaluate(now, sig)

          // Update Prometheus inflight gauges (sum across tenants AND per-tenant).
          obs.GatewayInflight.WithLabelValues(upstream).Set(float64(globalInflight))

          // Per-tenant inflight gauge fanout (Plan 03 Task 3.1 InflightRegistry.ObserveMetrics)
          // - emits gateway_inflight_tenant{upstream,tenant}. d.TenantLabel converts the uuid.UUID
          // tenant ID to its slug; pass nil to skip per-tenant emission (used in unit tests).
          if d.Inflight != nil && d.TenantLabel != nil {
              d.Inflight.ObserveMetrics(upstream, d.TenantLabel)
          }
      })
  }
  ```
