  ```go
  // Package dcgm (scraper.go): HTTP scraper for the pod-side dcgm-exporter
  // :9400/metrics endpoint. Runs as a single goroutine; publishes the VRAM
  // used (MiB) via atomic.Int64 for the FSM ticker to consume, and updates
  // the gateway_vram_used_mib Prometheus gauge for dashboards.
  //
  // Fail-open: 3 consecutive scrape failures flip vramUnknown=true, which
  // the FSM interprets as "ignore VRAM signal" (2-of-3 gate reduces to
  // 1-of-2 using inflight+P95). See CONTEXT.md D-A3 / RESEARCH.md Pitfall 1.
  //
  // Unit note: DCGM_FI_DEV_FB_USED is always in MiB per NVIDIA's
  // dcp-metrics-included.csv — never bytes. Do not convert.
  package dcgm

  import (
      "context"
      "fmt"
      "log/slog"
      "net/http"
      "sync/atomic"
      "time"

      dto "github.com/prometheus/client_model/go"
      "github.com/prometheus/common/expfmt"
      "github.com/prometheus/common/model"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
  )

  // Sanity bounds for parsed VRAM value (MiB). <0 is impossible; >1_000_000
  // (=~1TB VRAM) is far above any current GPU we ship — reject as tampering
  // or parsing glitch.
  const (
      minValidVramMiB = 0
      maxValidVramMiB = 1_000_000
      failOpenAfterN  = 3
  )

  type Scraper struct {
      url      string
      client   *http.Client
      log      *slog.Logger
      interval time.Duration

      vramUsedMiB     atomic.Int64
      vramUnknown     atomic.Bool
      consecutiveFail atomic.Int32
  }

  // New creates a Scraper. url must be a full URL (e.g. http://pod-host:9400/metrics).
  // interval is the tick period (default 5s); timeout bounds a single HTTP call
  // (default 2s).
  func New(url string, interval, timeout time.Duration, log *slog.Logger) *Scraper {
      if interval <= 0 {
          interval = 5 * time.Second
      }
      if timeout <= 0 {
          timeout = 2 * time.Second
      }
      return &Scraper{
          url: url,
          client: &http.Client{
              Timeout: timeout,
              Transport: &http.Transport{
                  DisableKeepAlives: false,
                  MaxIdleConns:      1,
                  IdleConnTimeout:   60 * time.Second,
              },
          },
          log:      log.With("module", "DCGM"),
          interval: interval,
      }
  }

  // Run blocks until ctx cancellation, scraping on each tick.
  // Intended to be called via `go scraper.Run(ctx)` from main.go.
  func (s *Scraper) Run(ctx context.Context) {
      t := time.NewTicker(s.interval)
      defer t.Stop()
      // Scrape immediately at boot so vramUsedMiB has a value before first FSM tick.
      s.scrape(ctx)
      for {
          select {
          case <-ctx.Done():
              s.log.Info("dcgm scraper stopping")
              return
          case <-t.C:
              s.scrape(ctx)
          }
      }
  }

  func (s *Scraper) scrape(ctx context.Context) {
      req, err := http.NewRequestWithContext(ctx, http.MethodGet, s.url, nil)
      if err != nil {
          s.fail("request_build", err)
          return
      }
      resp, err := s.client.Do(req)
      if err != nil {
          s.fail("http_error", err)
          return
      }
      defer resp.Body.Close()

      if resp.StatusCode != http.StatusOK {
          s.fail(fmt.Sprintf("status_%d", resp.StatusCode), nil)
          return
      }

      parser := expfmt.NewTextParser(model.UTF8Validation)
      families, err := parser.TextToMetricFamilies(resp.Body)
      if err != nil {
          s.fail("parse_error", err)
          return
      }

      fam, ok := families["DCGM_FI_DEV_FB_USED"]
      if !ok || len(fam.Metric) == 0 {
          s.fail("metric_missing", nil)
          return
      }

      var val float64
      if m := fam.Metric[0]; m.Gauge != nil {
          val = m.Gauge.GetValue()
      } else if fam.Metric[0].Counter != nil {
          // Defensive: fixture used counter instead of gauge
          val = fam.Metric[0].Counter.GetValue()
      } else {
          s.fail("metric_not_gauge", nil)
          return
      }

      if val < float64(minValidVramMiB) || val > float64(maxValidVramMiB) {
          s.fail("sanity_check", fmt.Errorf("vram value %.1f MiB out of bounds [%d, %d]",
              val, minValidVramMiB, maxValidVramMiB))
          return
      }

      s.vramUsedMiB.Store(int64(val))
      s.vramUnknown.Store(false)
      s.consecutiveFail.Store(0)
      obs.GatewayVramUsedMiB.Set(val)
      s.log.Debug("dcgm scrape ok", "vram_used_mib", val)
      _ = dto.MetricType(0) // keep dto imported even if Gauge branch is used
  }

  func (s *Scraper) fail(reason string, err error) {
      n := s.consecutiveFail.Add(1)
      obs.GatewayDcgmScrapeFailures.WithLabelValues(reason).Inc()
      if err != nil {
          s.log.Warn("dcgm scrape failed", "reason", reason, "err", err, "consecutive", n)
      } else {
          s.log.Warn("dcgm scrape failed", "reason", reason, "consecutive", n)
      }
      if n >= failOpenAfterN {
          s.vramUnknown.Store(true)
      }
  }

  // ReadMiB returns (vramUsedMiB, unknown). If unknown is true, the FSM
  // must treat the VRAM signal as not-contributing to the 2-of-3 gate.
  // Lockless — safe for hot path.
  func (s *Scraper) ReadMiB() (int64, bool) {
      if s == nil {
          // If no scraper was constructed (DCGM_EXPORTER_URL empty), treat as unknown.
          return 0, true
      }
      return s.vramUsedMiB.Load(), s.vramUnknown.Load()
  }
  ```

  import (

      srv := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, _ *http.Request) {

          _, _ = w.Write([]byte(validMetricsBody))
