<objective>
Ship `gatewayctl audit export-month YYYY-MM` so operators can offload audit_log partitions older than 90 days to MinIO Parquet (CONTEXT.md D-B3: "90 dias hot no DO Postgres + 1 ano cold em MinIO Ifix"). After upload succeeds, the corresponding partition is dropped from Postgres. Everything is idempotent, non-blocking, and safe (never drops an un-uploaded partition).

Purpose: Close the retention contract implied by GW-10's schema design (partitioning was chosen so partitions could be dropped wholesale). Provides the operator with a tested CLI invocation NOW; Phase 4+ wraps it in Kestra cron.

Output: 1 gatewayctl subcommand file + 3 coldstore source files + 3 tests + config.Config extension for MinIO env vars + README update documenting the command.

```go
// Package coldstore implements Parquet-to-MinIO export of audit_log
// partitions + safe drop of partitions outside the retention window
// (CONTEXT.md D-B3: 90 days hot / 1 year cold).
package coldstore

import (
    "fmt"
    "time"
)

// IsBeyondRetentionWindow returns true when the LAST day of `month` is
// more than minAgeDays before `now` — i.e., every row in that month
// partition is older than the retention window.
func IsBeyondRetentionWindow(month time.Time, now time.Time, minAgeDays int) bool {
    // End of the month (first day of next month minus 1 nanosecond).
    startOfNext := time.Date(month.Year(), month.Month()+1, 1, 0, 0, 0, 0, month.Location())
    ageDays := int(now.Sub(startOfNext).Hours() / 24)
    return ageDays >= minAgeDays
}

// PartitionName returns the fully-qualified partition table name for a
// given month, matching the naming convention from migration 0003/0004
// (e.g., ai_gateway.audit_log_202604).
func PartitionName(schema, baseTable string, month time.Time) string {
    return fmt.Sprintf("%s.%s_%s", schema, baseTable, month.Format("200601"))
}

// BucketKeyForMeta returns the MinIO object key for the metadata parquet.
func BucketKeyForMeta(month time.Time) string {
    return fmt.Sprintf("%04d/%02d/audit_log_%s.parquet",
        month.Year(), int(month.Month()), month.Format("200601"))
}

// BucketKeyForContent returns the MinIO object key for the content parquet.
func BucketKeyForContent(month time.Time) string {
    return fmt.Sprintf("%04d/%02d/audit_log_content_%s.parquet",
        month.Year(), int(month.Month()), month.Format("200601"))
}
```

```go
package coldstore

import (
    "context"
    "fmt"
    "io"

    "github.com/minio/minio-go/v7"
    "github.com/minio/minio-go/v7/pkg/credentials"

    "github.com/ifixtelecom/gpu-ifix/gateway/internal/config"
)

// Client wraps the minio-go client for our cold-storage operations.
type Client struct {
    m      *minio.Client
    bucket string
}

// NewClient connects to the MinIO endpoint and verifies bucket existence
// (creates it if missing — one-time operator provisioning is fine for
// dev; prod may lock down bucket creation via IAM).
func NewClient(ctx context.Context, cfg config.Config) (*Client, error) {
    if m := cfg.MissingMinIOVars(); len(m) > 0 {
        return nil, fmt.Errorf("coldstore: missing MinIO env vars: %v", m)
    }
    mc, err := minio.New(cfg.MinIOEndpoint, &minio.Options{
        Creds:  credentials.NewStaticV4(cfg.MinIOAccessKey, cfg.MinIOSecretKey, ""),
        Secure: cfg.MinIOUseSSL,
    })
    if err != nil { return nil, fmt.Errorf("coldstore: new minio client: %w", err) }

    exists, err := mc.BucketExists(ctx, cfg.MinIOBucket)
    if err != nil { return nil, fmt.Errorf("coldstore: bucket exists check: %w", err) }
    if !exists {
        if err := mc.MakeBucket(ctx, cfg.MinIOBucket, minio.MakeBucketOptions{}); err != nil {
            return nil, fmt.Errorf("coldstore: create bucket %q: %w", cfg.MinIOBucket, err)
        }
    }
    return &Client{m: mc, bucket: cfg.MinIOBucket}, nil
}

// Put uploads `data` with `size` bytes to bucket/key. Overwrites on re-run.
func (c *Client) Put(ctx context.Context, key string, data io.Reader, size int64, contentType string) error {
    _, err := c.m.PutObject(ctx, c.bucket, key, data, size, minio.PutObjectOptions{
        ContentType: contentType,
    })
    if err != nil { return fmt.Errorf("coldstore: put %s: %w", key, err) }
    return nil
}

// Exists returns whether the object is present in the bucket.
func (c *Client) Exists(ctx context.Context, key string) (bool, error) {
    _, err := c.m.StatObject(ctx, c.bucket, key, minio.StatObjectOptions{})
    if err != nil {
        e := minio.ToErrorResponse(err)
        if e.Code == "NoSuchKey" { return false, nil }
        return false, err
    }
    return true, nil
}
```

```go
package coldstore

import (
    "context"
    "io"
    "time"

    "github.com/google/uuid"
    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgxpool"
    "github.com/parquet-go/parquet-go"
)

// AuditLogRow mirrors ai_gateway.audit_log columns. Field names/JSON tags
// align with the Postgres column names for human legibility when the
// parquet is read back via DuckDB / Athena / DataFrames. parquet tags
// configure column encoding.
type AuditLogRow struct {
    Ts                  time.Time `parquet:"ts,timestamp(millisecond)"`
    RequestID           []byte    `parquet:"request_id,fixed(16)"` // UUID encoded as 16 bytes
    TenantID            []byte    `parquet:"tenant_id,fixed(16)"`
    APIKeyID            []byte    `parquet:"api_key_id,optional,fixed(16)"`
    DataClass           string    `parquet:"data_class,dict,zstd"`
    Route               string    `parquet:"route,dict,zstd"`
    Method              string    `parquet:"method,dict,zstd"`
    Upstream            *string   `parquet:"upstream,optional"`
    StatusCode          int32     `parquet:"status_code"`
    LatencyMs           int32     `parquet:"latency_ms"`
    TokensIn            *int32    `parquet:"tokens_in,optional"`
    TokensOut           *int32    `parquet:"tokens_out,optional"`
    CostBRL             *float64  `parquet:"cost_brl,optional"`
    ErrorCode           *string   `parquet:"error_code,optional"`
    IdempotencyReplayed bool      `parquet:"idempotency_replayed"`
    Stream              bool      `parquet:"stream"`
    Truncated           bool      `parquet:"truncated"`
    AudioFilename       *string   `parquet:"audio_filename,optional"`
    AudioMime           *string   `parquet:"audio_mime,optional"`
    AudioSizeBytes      *int64    `parquet:"audio_size_bytes,optional"`
    AudioDurationS      *float32  `parquet:"audio_duration_s,optional"`
    AudioLanguage       *string   `parquet:"audio_language,optional"`
}

// AuditLogContentRow mirrors ai_gateway.audit_log_content.
type AuditLogContentRow struct {
    RequestID []byte    `parquet:"request_id,fixed(16)"`
    Ts        time.Time `parquet:"ts,timestamp(millisecond)"`
    Prompt    []byte    `parquet:"prompt,optional,zstd"`   // JSONB stored as raw bytes
    Response  []byte    `parquet:"response,optional,zstd"`
}

// WriteAuditLog streams SELECT ... FROM <partition> into a Parquet writer.
// Returns the number of rows written.
func WriteAuditLog(ctx context.Context, pool *pgxpool.Pool, partition string, dst io.Writer) (int, error) {
    w := parquet.NewGenericWriter[AuditLogRow](dst)
    defer w.Close()

    rows, err := pool.Query(ctx, "SELECT * FROM "+partition+" ORDER BY ts")
    if err != nil { return 0, err }
    defer rows.Close()

    var written int
    batch := make([]AuditLogRow, 0, 500)
    for rows.Next() {
        r, err := scanAuditLogRow(rows)
        if err != nil { return written, err }
        batch = append(batch, r)
        if len(batch) >= 500 {
            if _, err := w.Write(batch); err != nil { return written, err }
            written += len(batch)
            batch = batch[:0]
        }
    }
    if len(batch) > 0 {
        if _, err := w.Write(batch); err != nil { return written, err }
        written += len(batch)
    }
    if err := rows.Err(); err != nil { return written, err }
    return written, w.Close()
}

// WriteAuditLogContent likewise.
func WriteAuditLogContent(ctx context.Context, pool *pgxpool.Pool, partition string, dst io.Writer) (int, error) {
    w := parquet.NewGenericWriter[AuditLogContentRow](dst)
    defer w.Close()
    rows, err := pool.Query(ctx, "SELECT request_id, ts, prompt, response FROM "+partition+" ORDER BY ts")
    if err != nil { return 0, err }
    defer rows.Close()
    var written int
    batch := make([]AuditLogContentRow, 0, 500)
    for rows.Next() {
        var (
            rid uuid.UUID
            ts time.Time
            prompt, resp []byte
        )
        if err := rows.Scan(&rid, &ts, &prompt, &resp); err != nil { return written, err }
        batch = append(batch, AuditLogContentRow{
            RequestID: rid[:], Ts: ts, Prompt: prompt, Response: resp,
        })
        if len(batch) >= 500 {
            if _, err := w.Write(batch); err != nil { return written, err }
            written += len(batch)
            batch = batch[:0]
        }
    }
    if len(batch) > 0 {
        if _, err := w.Write(batch); err != nil { return written, err }
        written += len(batch)
    }
    if err := rows.Err(); err != nil { return written, err }
    return written, w.Close()
}

// scanAuditLogRow is a helper that reads all audit_log columns into the
// Parquet-shaped AuditLogRow. Uses explicit column order matching
// migration 0003_create_audit_log_partitioned.sql.
func scanAuditLogRow(rows pgx.Rows) (AuditLogRow, error) {
    var r AuditLogRow
    var reqID, tenantID uuid.UUID
    var apiKeyID *uuid.UUID
    var upstream, errorCode, audioFilename, audioMime, audioLang *string
    var tokensIn, tokensOut *int32
    var costBRL *float64
    var audioSize *int64
    var audioDur *float32
    var dataClass, route, method string

    if err := rows.Scan(
        &r.Ts, &reqID, &tenantID, &apiKeyID, &dataClass,
        &route, &method, &upstream, &r.StatusCode, &r.LatencyMs,
        &tokensIn, &tokensOut, &costBRL, &errorCode,
        &r.IdempotencyReplayed, &r.Stream, &r.Truncated,
        &audioFilename, &audioMime, &audioSize, &audioDur, &audioLang,
    ); err != nil {
        return AuditLogRow{}, err
    }
    r.RequestID = reqID[:]
    r.TenantID = tenantID[:]
    if apiKeyID != nil { r.APIKeyID = (*apiKeyID)[:] }
    r.DataClass = dataClass
    r.Route = route
    r.Method = method
    r.Upstream = upstream
    r.TokensIn = tokensIn
    r.TokensOut = tokensOut
    r.CostBRL = costBRL
    r.ErrorCode = errorCode
    r.AudioFilename = audioFilename
    r.AudioMime = audioMime
    r.AudioSizeBytes = audioSize
    r.AudioDurationS = audioDur
    r.AudioLanguage = audioLang
    return r, nil
}
```

```go
package main

import (
    "bytes"
    "context"
    "encoding/json"
    "flag"
    "fmt"
    "log/slog"
    "time"

    "github.com/ifixtelecom/gpu-ifix/gateway/internal/coldstore"
)

type exportReport struct {
    Month              string `json:"month"`
    ExportedMetaRows   int    `json:"exported_meta_rows"`
    ExportedContentRows int   `json:"exported_content_rows"`
    ParquetSizeBytes   int64  `json:"parquet_size_bytes"`
    BucketPathMeta     string `json:"bucket_path_meta"`
    BucketPathContent  string `json:"bucket_path_content"`
    PartitionDropped   bool   `json:"partition_dropped"`
    Reason             string `json:"reason,omitempty"`
    DryRun             bool   `json:"dry_run"`
}

func runAudit(ctx context.Context, args []string, log *slog.Logger) int {
    if len(args) == 0 {
        fmt.Fprintln(flag.CommandLine.Output(), "Usage: gatewayctl audit export-month --month YYYY-MM [--dry-run]")
        return 2
    }
    switch args[0] {
    case "export-month":
        return runAuditExportMonth(ctx, args[1:], log)
    default:
        fmt.Fprintf(flag.CommandLine.Output(), "unknown audit subcommand: %s\n", args[0])
        return 2
    }
}

func runAuditExportMonth(ctx context.Context, args []string, log *slog.Logger) int {
    fs := flag.NewFlagSet("audit export-month", flag.ExitOnError)
    monthStr := fs.String("month", "", "YYYY-MM partition to export (required)")
    dryRun := fs.Bool("dry-run", false, "Preview without uploading or dropping")
    if err := fs.Parse(args); err != nil { return 2 }

    if *monthStr == "" {
        fs.Usage()
        return 2
    }
    month, err := time.Parse("2006-01", *monthStr)
    if err != nil {
        fmt.Fprintf(fs.Output(), "error: month must be YYYY-MM: %v\n", err)
        return 2
    }

    cfg, pool, err := loadAndPool(ctx, log)
    if err != nil {
        fmt.Fprintf(fs.Output(), "error: %v\n", err)
        return 1
    }
    defer pool.Close()

    if missing := cfg.MissingMinIOVars(); len(missing) > 0 {
        fmt.Fprintf(fs.Output(), "error: MinIO env vars not set: %v\n", missing)
        return 2
    }

    client, err := coldstore.NewClient(ctx, cfg)
    if err != nil {
        fmt.Fprintf(fs.Output(), "error: minio client: %v\n", err)
        return 1
    }

    metaPart := coldstore.PartitionName("ai_gateway", "audit_log", month)
    contentPart := coldstore.PartitionName("ai_gateway", "audit_log_content", month)
    keyMeta := coldstore.BucketKeyForMeta(month)
    keyContent := coldstore.BucketKeyForContent(month)

    report := exportReport{
        Month:             month.Format("2006-01"),
        BucketPathMeta:    keyMeta,
        BucketPathContent: keyContent,
        DryRun:            *dryRun,
    }

    // Export metadata.
    var metaBuf bytes.Buffer
    n, err := coldstore.WriteAuditLog(ctx, pool, metaPart, &metaBuf)
    if err != nil {
        fmt.Fprintf(fs.Output(), "error: write audit_log parquet: %v\n", err)
        return 1
    }
    report.ExportedMetaRows = n

    // Export content.
    var contentBuf bytes.Buffer
    n, err = coldstore.WriteAuditLogContent(ctx, pool, contentPart, &contentBuf)
    if err != nil {
        fmt.Fprintf(fs.Output(), "error: write audit_log_content parquet: %v\n", err)
        return 1
    }
    report.ExportedContentRows = n
    report.ParquetSizeBytes = int64(metaBuf.Len() + contentBuf.Len())

    if *dryRun {
        log.Info("dry-run — no upload, no drop", "report", report)
        jsonPrint(report)
        return 0
    }

    // Upload.
    if err := client.Put(ctx, keyMeta, &metaBuf, int64(metaBuf.Len()), "application/octet-stream"); err != nil {
        fmt.Fprintf(fs.Output(), "error: upload meta parquet: %v\n", err)
        return 1
    }
    if err := client.Put(ctx, keyContent, &contentBuf, int64(contentBuf.Len()), "application/octet-stream"); err != nil {
        fmt.Fprintf(fs.Output(), "error: upload content parquet: %v\n", err)
        return 1
    }

    // Drop partition ONLY if outside retention window.
    if coldstore.IsBeyondRetentionWindow(month, time.Now(), cfg.AuditRetentionDays) {
        // DROP both partitions in a single tx.
        tx, err := pool.Begin(ctx)
        if err != nil {
            fmt.Fprintf(fs.Output(), "error: begin tx: %v\n", err)
            return 1
        }
        if _, err := tx.Exec(ctx, "DROP TABLE IF EXISTS "+metaPart); err != nil {
            _ = tx.Rollback(ctx)
            fmt.Fprintf(fs.Output(), "error: drop %s: %v\n", metaPart, err)
            return 1
        }
        if _, err := tx.Exec(ctx, "DROP TABLE IF EXISTS "+contentPart); err != nil {
            _ = tx.Rollback(ctx)
            fmt.Fprintf(fs.Output(), "error: drop %s: %v\n", contentPart, err)
            return 1
        }
        if err := tx.Commit(ctx); err != nil {
            fmt.Fprintf(fs.Output(), "error: commit: %v\n", err)
            return 1
        }
        report.PartitionDropped = true
    } else {
        report.PartitionDropped = false
        report.Reason = "month_within_retention_window"
    }

    log.Info("audit export-month complete",
        "month", report.Month,
        "meta_rows", report.ExportedMetaRows,
        "content_rows", report.ExportedContentRows,
        "dropped", report.PartitionDropped,
    )
    jsonPrint(report)
    return 0
}

func jsonPrint(r exportReport) {
    b, _ := json.MarshalIndent(r, "", "  ")
    fmt.Println(string(b))
}
```
