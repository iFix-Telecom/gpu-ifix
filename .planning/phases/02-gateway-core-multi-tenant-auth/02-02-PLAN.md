```sql
-- +goose Up
-- +goose StatementBegin
SET search_path = ai_gateway, public;

DO $$ BEGIN
    CREATE TYPE ai_gateway.api_key_status AS ENUM ('active', 'revoked');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
    CREATE TYPE ai_gateway.data_class AS ENUM ('normal', 'sensitive');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

CREATE TABLE IF NOT EXISTS ai_gateway.api_keys (
    id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id    UUID NOT NULL REFERENCES ai_gateway.tenants(id) ON DELETE CASCADE,
    key_hash     TEXT NOT NULL,
    key_prefix   TEXT NOT NULL,
    status       ai_gateway.api_key_status NOT NULL DEFAULT 'active',
    data_class   ai_gateway.data_class NOT NULL,
    created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    revoked_at   TIMESTAMPTZ,
    last_used_at TIMESTAMPTZ
);

CREATE INDEX IF NOT EXISTS idx_api_keys_tenant           ON ai_gateway.api_keys(tenant_id);
CREATE INDEX IF NOT EXISTS idx_api_keys_status_tenant    ON ai_gateway.api_keys(status, tenant_id);
CREATE INDEX IF NOT EXISTS idx_api_keys_key_prefix       ON ai_gateway.api_keys(key_prefix);
-- +goose StatementEnd

-- +goose Down
-- +goose StatementBegin
DROP TABLE IF EXISTS ai_gateway.api_keys;
DROP TYPE IF EXISTS ai_gateway.api_key_status;
DROP TYPE IF EXISTS ai_gateway.data_class;
-- +goose StatementEnd
```

FILE `gateway/db/migrations/0003_create_audit_log_partitioned.sql`:

```sql
-- +goose Up
-- +goose StatementBegin
SET search_path = ai_gateway, public;

CREATE TABLE IF NOT EXISTS ai_gateway.audit_log (
    ts                    TIMESTAMPTZ NOT NULL,
    request_id            UUID NOT NULL,
    tenant_id             UUID NOT NULL,
    api_key_id            UUID,
    data_class            ai_gateway.data_class NOT NULL,
    route                 TEXT NOT NULL,
    method                TEXT NOT NULL,
    upstream              TEXT,
    status_code           SMALLINT NOT NULL,
    latency_ms            INTEGER NOT NULL,
    tokens_in             INTEGER,
    tokens_out            INTEGER,
    cost_brl              NUMERIC(10, 4),
    error_code            TEXT,
    idempotency_replayed  BOOLEAN NOT NULL DEFAULT FALSE,
    stream                BOOLEAN NOT NULL DEFAULT FALSE,
    truncated             BOOLEAN NOT NULL DEFAULT FALSE,
    -- Whisper metadata (D-B6 — audio stays out of content table but
    -- envelope metadata is safe to persist)
    audio_filename        TEXT,
    audio_mime            TEXT,
    audio_size_bytes      BIGINT,
    audio_duration_s      REAL,
    audio_language        TEXT,
    PRIMARY KEY (request_id, ts)
) PARTITION BY RANGE (ts);

CREATE INDEX IF NOT EXISTS idx_audit_log_tenant_ts  ON ai_gateway.audit_log(tenant_id, ts DESC);
CREATE INDEX IF NOT EXISTS idx_audit_log_request_id ON ai_gateway.audit_log(request_id);
CREATE INDEX IF NOT EXISTS idx_audit_log_api_key    ON ai_gateway.audit_log(api_key_id) WHERE api_key_id IS NOT NULL;

-- Seed partitions for the current month and the next 2 months so that
-- the first inserts land somewhere. Plan 02-09's gatewayctl command
-- rolls new partitions forward and drops ones older than 90 days.
DO $$
DECLARE
    m DATE;
    start_m DATE;
    end_m DATE;
    part_name TEXT;
BEGIN
    FOR i IN 0..2 LOOP
        m := DATE_TRUNC('month', CURRENT_DATE) + (i || ' months')::INTERVAL;
        start_m := m;
        end_m := m + INTERVAL '1 month';
        part_name := 'audit_log_' || to_char(start_m, 'YYYYMM');
        EXECUTE format(
            'CREATE TABLE IF NOT EXISTS ai_gateway.%I PARTITION OF ai_gateway.audit_log FOR VALUES FROM (%L) TO (%L)',
            part_name, start_m, end_m
        );
    END LOOP;
END $$;
-- +goose StatementEnd

-- +goose Down
-- +goose StatementBegin
DROP TABLE IF EXISTS ai_gateway.audit_log CASCADE;
-- +goose StatementEnd
```

```markdown
- Hot (Postgres DO): 90 dias (D-B3).
- Cold (MinIO `ifix-ai-gateway-audit-cold`): 1 ano, Parquet mensal.
- Export + drop de partições >90d: `gatewayctl audit export-month YYYY-MM`

- `audit_log` e `audit_log_content`: `PARTITION BY RANGE (ts)`, PK composto
  `(request_id, ts)` (Postgres exige que a coluna de particionamento esteja

## Out of scope (Fase 2)

- Admin REST endpoints → Fase 7 (dashboard Next.js).
```

```sql
-- name: ListActiveKeysByTenant :many
-- Used by auth.Verifier when we know the tenant (e.g. migration diagnostic).
-- NOT on the hot request path — that goes via ListActiveKeysAll + in-memory
-- argon2 verify with cache lookup beforehand.
SELECT id, tenant_id, key_hash, key_prefix, status, data_class, created_at, revoked_at, last_used_at
FROM ai_gateway.api_keys
WHERE tenant_id = $1 AND status = 'active'
ORDER BY last_used_at DESC NULLS LAST;

-- name: ListActiveKeysAll :many
-- Hot path lookup: returns every active key row. Gateway compares argon2
-- hash in Go. With small total key counts (~10 keys per tenant × <10
-- tenants) this is O(100) comparisons; Redis cache handles subsequent
-- hits (CONTEXT.md D-A2). Result rows carry tenant_id so the caller
-- does not need a second lookup.
SELECT id, tenant_id, key_hash, key_prefix, status, data_class
FROM ai_gateway.api_keys
WHERE status = 'active';

-- name: GetAPIKeyByID :one
SELECT id, tenant_id, key_hash, key_prefix, status, data_class, created_at, revoked_at, last_used_at
FROM ai_gateway.api_keys
WHERE id = $1;

-- name: TouchKeyLastUsed :exec
UPDATE ai_gateway.api_keys SET last_used_at = NOW() WHERE id = $1;
```

```go
// Package db owns the gateway's Postgres connection pool and migration
// runner. Every package that talks to Postgres goes through the pgxpool
// created here. Schema isolation (D-D4) is enforced via an AfterConnect
// hook that sets search_path on every acquired connection.
package db

import (
	"context"
	"fmt"
	"time"

	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgxpool"

	"github.com/ifixtelecom/gpu-ifix/gateway/internal/config"
)

// NewPool opens a pgxpool.Pool against cfg.PGDSN with search_path hooked
// to 'ai_gateway, public' on every connection acquired.
func NewPool(ctx context.Context, cfg config.Config) (*pgxpool.Pool, error) {
	pcfg, err := pgxpool.ParseConfig(cfg.PGDSN)
	if err != nil {
		return nil, fmt.Errorf("db: parse DSN: %w", err)
	}
	if cfg.PGMaxConns > 0 {
		pcfg.MaxConns = cfg.PGMaxConns
	} else {
		pcfg.MaxConns = 10
	}
	pcfg.MaxConnIdleTime = 5 * time.Minute
	pcfg.AfterConnect = func(ctx context.Context, conn *pgx.Conn) error {
		_, err := conn.Exec(ctx, "SET search_path = ai_gateway, public")
		if err != nil {
			return fmt.Errorf("db: set search_path: %w", err)
		}
		return nil
	}

	pool, err := pgxpool.NewWithConfig(ctx, pcfg)
	if err != nil {
		return nil, fmt.Errorf("db: new pool: %w", err)
	}
	// Fail-fast ping so a bad DSN surfaces at startup, not on first request.
	if err := pool.Ping(ctx); err != nil {
		pool.Close()
		return nil, fmt.Errorf("db: ping: %w", err)
	}
	return pool, nil
}
```

FILE `gateway/internal/db/migrate.go`:

```go
package db

import (
	"context"
	"database/sql"
	"embed"
	"fmt"

	"github.com/jackc/pgx/v5/pgxpool"
	"github.com/jackc/pgx/v5/stdlib"
	"github.com/pressly/goose/v3"
)

//go:embed migrations/*.sql
var migrationsFS embed.FS

func newGooseDB(pool *pgxpool.Pool) (*sql.DB, error) {
	// goose requires a *sql.DB; pgx provides an adapter.
	return stdlib.OpenDBFromPool(pool), nil
}

// Up applies all pending migrations in order. Idempotent — goose records
// the applied version in ai_gateway.goose_db_version. search_path is
// forced to ai_gateway,public by the pool's AfterConnect hook, so goose
// creates its bookkeeping table under the same schema.
func Up(ctx context.Context, pool *pgxpool.Pool) error {
	db, err := newGooseDB(pool)
	if err != nil {
		return err
	}
	defer db.Close()

	goose.SetBaseFS(migrationsFS)
	if err := goose.SetDialect("postgres"); err != nil {
		return fmt.Errorf("db.Up: set dialect: %w", err)
	}
	if err := goose.UpContext(ctx, db, "migrations"); err != nil {
		return fmt.Errorf("db.Up: %w", err)
	}
	return nil
}

// Down rolls back n migrations (use 0 for all).
func Down(ctx context.Context, pool *pgxpool.Pool, n int) error {
	db, err := newGooseDB(pool)
	if err != nil {
		return err
	}
	defer db.Close()
	goose.SetBaseFS(migrationsFS)
	if err := goose.SetDialect("postgres"); err != nil {
		return err
	}
	for i := 0; i < n; i++ {
		if err := goose.DownContext(ctx, db, "migrations"); err != nil {
			return fmt.Errorf("db.Down[%d]: %w", i, err)
		}
	}
	return nil
}

// Status prints the version table to stdout (used by `gatewayctl migrate status`).
func Status(ctx context.Context, pool *pgxpool.Pool) error {
	db, err := newGooseDB(pool)
	if err != nil {
		return err
	}
	defer db.Close()
	goose.SetBaseFS(migrationsFS)
	if err := goose.SetDialect("postgres"); err != nil {
		return err
	}
	return goose.StatusContext(ctx, db, "migrations")
}
```

```go
	entries, err := fs.ReadDir(migrationsFS, "migrations")
	if err != nil {
		t.Fatal(err)
	}
	var names []string
	for _, e := range entries {
		if strings.HasSuffix(e.Name(), ".sql") {
			names = append(names, e.Name())
		}

	sort.Strings(names)

		"0001_create_tenants.sql",
		"0002_create_api_keys.sql",
		"0003_create_audit_log_partitioned.sql",
		"0004_create_audit_log_content_partitioned.sql",
		"0005_create_model_aliases.sql",
		"0006_create_usage_counters_skeleton.sql",
	}
	for i, w := range want {
		if names[i] != w {
			t.Errorf("[%d] got %s want %s", i, names[i], w)
		}
	}
}
```

## Trust Boundaries

| Boundary | Description |
|---|---|
| Gateway/gatewayctl binary → DO Postgres | TLS-only DSN expected (`sslmode=require` in production); gateway binary is the only consumer besides manual `psql` operator access |
| Postgres role `ai_gateway_app` → schema `ai_gateway` | GRANT limited to this schema; no access to other Ifix app schemas on the shared cluster |
| Goose migration file → Postgres | Files are signed implicitly by git commit; deploy pipeline (Plan 02-08) is the only execution path |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|---|---|---|---|---|---|
| T-02-02-01 | Elevation of Privilege | Role isolation | high | mitigate | DSN uses `ai_gateway_app` role (not superuser); migrations assume schema `ai_gateway` exists and role has GRANT USAGE. Operator provisioning documented in `gateway/db/README.md`; migration files do NOT attempt `CREATE ROLE` or `GRANT` (operator responsibility on the shared DO cluster) |
| T-02-02-02 | Tampering | Cross-schema writes | medium | mitigate | `search_path = ai_gateway, public` on every connection via pgxpool AfterConnect; queries use `ai_gateway.table_name` fully qualified as defense-in-depth; sqlc-generated code inherits qualified names |
| T-02-02-03 | Information Disclosure | Postgres logs leaking query parameters | low | accept | All statements use `$1 $2 ...` placeholders (sqlc default); Postgres log settings on DO cluster are the operator's responsibility; gateway does not log the raw DSN above INFO level |
| T-02-02-04 | DoS | Migration running goose under load | low | mitigate | Goose wraps each migration in a transaction; failure rolls back; idempotent on rerun. `gatewayctl migrate` is a manual admin operation, not on hot request path |
| T-02-02-05 | Information Disclosure | Audit content partition dropped contains sensitive PII for 90d | high | mitigate | Row is only inserted for `data_class = 'normal'` (Plan 02-05 writer enforces; schema permits but code filters). `audit_log_content` has NO FK to keep sensitive tenants strictly absent. Parquet export (Plan 02-09) respects same filter. Retention 90d then drop of partition = physical removal |
| T-02-02-06 | DoS | Unbounded partition growth | low | mitigate | Plan 02-09 ships `gatewayctl audit` which drops partitions older than 90 days after Parquet export. Phase 4 promotes to cron |

Note: T-02-02-05 is downstream-enforced (writer code). Schema here permits any insert; the writer's filter is the security control.
