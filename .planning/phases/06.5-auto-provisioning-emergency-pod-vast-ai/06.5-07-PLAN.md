       func (r *Reconciler) recoverOrphanLifecycles(ctx context.Context) error {
           rows, err := r.q.ListLiveEmergencyLifecycles(ctx)
           if err != nil { return err }
           for _, row := range rows {

       // BLOCKER 4 (revisão 2026-05-13) — FUNCIONAL JSONB events replay; NÃO placeholder.
       // Walks events array para reconstruir state preciso; re-attach cancel context;
       // restart healthcheck goroutine se EMERGENCY_ACTIVE; override dispatcher tier-0.
       func (r *Reconciler) resumeFSMFromEvents(parentCtx context.Context, row gen.ListLiveEmergencyLifecyclesRow, inst vast.Instance) error {
           // 1. Parse events JSONB para determinar last to_state
           var events []map[string]any
           if err := json.Unmarshal(row.Events, &events); err != nil {
               return fmt.Errorf("unmarshal events: %w", err)
           }
           lastState := StateEmergencyActive  // default sane assumption (instance is running per inst.IsActive())
           if len(events) > 0 {
               last := events[len(events)-1]
               if to, ok := last["to_state"].(string); ok {
                   if parsed, perr := ParseState(to); perr == nil {
                       lastState = parsed
                   }
               }
           }
           // 2. Atomic CAS via FSM.SetState (helper a adicionar em fsm.go — Plan 03 only has Transition)
           //    SetState is recovery-only escape hatch; not exposed to normal callers.
           r.deps.FSM.SetState(lastState, time.Now(), "leader_recovery_resume:"+strconv.FormatInt(row.ID, 10))
           // 3. Reconstruct activeLifecycle pointer com NEW cancel context
           ctx, cancel := context.WithCancel(parentCtx)
           r.activeLifecycle.Store(&Lifecycle{ID: row.ID, Cancel: cancel})
           // 4. Re-attach pod URL: derive from inst per spike outcome (06-SPIKE-vast-port-mapping.md)
           podURL := r.podHealthURL(inst)  // helper from Plan 06; safe to call here (inst.IsActive() == true)
           r.activePodURL.Store(&podURL)
           // 5. Override dispatcher tier-0 (D-E3 — Plan 08 helper)
           r.deps.Loader.OverrideTier0("llm", podURL)
           obs.GatewayEmergencyActivePod.WithLabelValues(podURL).Set(1)
           // 6. Restart healthcheck polling goroutine (D-A4 cold-start budget RE-USED with elapsed-time check)
           //    Conservative: only restart if state recovered == EMERGENCY_ACTIVE; if recovered == EMERGENCY_PROVISIONING,
           //    delegate to next reconciler tick which calls evaluateEmergencyProvisioning → startProvisioning detects
           //    activeLifecycle != nil + DB row exists, takes a "resume" branch instead of "fresh insert".
           if lastState == StateEmergencyActive {
               go r.runHealthcheckResumeLoop(ctx, row.ID, podURL)
           } else if lastState == StateEmergencyProvisioning {
               // Cold-start budget remaining: from started_at deadline. Provisioning continuation handled by next tick.
               r.deps.Log.Info("recovered EMERGENCY_PROVISIONING; next tick will continue health polling", "id", row.ID)
           }
           r.deps.Log.Info("FSM resumed from events JSONB", "id", row.ID, "state", lastState.String(), "pod_url", podURL, "events_count", len(events))
           return nil
       }

       // runHealthcheckResumeLoop — restart /health polling for resumed lifecycle.
       // Cancels lifecycle if /health fails for >3 consecutive checks (5s interval = 15s degradation budget).
       func (r *Reconciler) runHealthcheckResumeLoop(ctx context.Context, lifecycleID int64, podURL string) {
           failures := 0
           ticker := time.NewTicker(5 * time.Second)
           defer ticker.Stop()
           for {
               select {
               case <-ctx.Done(): return
               case <-ticker.C:
                   if r.checkHealth(ctx, podURL) == nil {
                       failures = 0
                       continue
                   }
                   failures++
                   if failures >= 3 {
                       r.deps.Log.Warn("resumed lifecycle /health failing; cancelling", "id", lifecycleID, "consecutive_failures", failures)
                       r.cancelActiveLifecycle(ctx, "resume_health_failed")
                       return
                   }
               }
           }
       }
