       type Lifecycle struct {
           ID     int64
           Cancel context.CancelFunc
       }

       func (r *Reconciler) startProvisioning(parentCtx context.Context) {
           // Insert lifecycle row primeiro (Pitfall 7 — cobrir orphan recovery em Plan 07)
           id, err := r.q.InsertEmergencyLifecycle(parentCtx, gen.InsertEmergencyLifecycleParams{
               TriggerReason: "failed_over_sustained",
               LeaderReplica: pgtype.Text{String: r.replicaID, Valid: true},
           })
           if err != nil {
               r.deps.Log.Error("InsertEmergencyLifecycle failed", "err", err)
               return
           }
           ctx, cancel := context.WithCancel(parentCtx)
           r.activeLifecycle.Store(&Lifecycle{ID: id, Cancel: cancel})
           go func() {
               defer cancel()
               start := time.Now()
               if err := r.provisionLifecycle(ctx, id); err != nil {
                   r.deps.Log.Error("provision lifecycle failed", "id", id, "err", err)
                   // FSM volta para Healthy se erro (Plan 07 refina semântica)
                   r.deps.FSM.Transition(StateEmergencyProvisioning, StateHealthy, time.Now(), "provision_error:"+err.Error())
               }
               obs.GatewayEmergencyProvisionDurationSeconds.Observe(time.Since(start).Seconds())
           }()
       }

       func (r *Reconciler) provisionLifecycle(ctx context.Context, id int64) error {
           if ctx.Err() != nil { return ctx.Err() }

           offers, err := r.deps.Vast.SearchOffers(ctx, vast.DefaultSearchFilter(r.deps.Cfg.VastPriceCapDPH))
           if err != nil { return err }

           // Pitfall 5 — epsilon when comparing dph_total to cap (defesa redundante: filter ja aplica via API server-side, mas operator pode override)
           pickable := filterBelowCap(offers, r.deps.Cfg.VastPriceCapDPH)
           // D-A2 — exclude PRIMARY_HOST_ID se conhecido
           if r.deps.Cfg.PrimaryHostID > 0 {
               pickable = excludeHost(pickable, r.deps.Cfg.PrimaryHostID)
           }
           if len(pickable) == 0 { return ErrNoOffersBelowCap }

           // D-A3 — 3 attempts bid race retry
           for attempt := 0; attempt < 3; attempt++ {
               if ctx.Err() != nil { return ctx.Err() }
               offer := pickable[0]
               req := r.createRequest(offer, id)
               instance, err := r.deps.Vast.CreateInstance(ctx, offer.ID, req)
               if err == nil {
                   // Update DB com vast IDs (D-D3: events JSONB written FIRST in same UPDATE — W7 revisão 2026-05-13)
                   eventJSON := mustEventJSON("offer_accepted", map[string]any{"offer_id": offer.ID, "dph": offer.DphTotal})
                   _ = r.q.UpdateEmergencyLifecycleVastIDs(ctx, gen.UpdateEmergencyLifecycleVastIDsParams{
                       ID: id, VastOfferID: pgInt8(offer.ID), VastInstanceID: pgInt8(instance.ID),
                       AcceptedDph: pgNumeric(offer.DphTotal), EventJson: eventJSON,
                   })
                   return r.waitForReadyOrDestroy(ctx, id, instance.ID)
               }
               if !errors.Is(err, vast.ErrOfferGone) { return err }
               // Bid race: re-search com 2s/4s/8s exp backoff
               select {
               case <-ctx.Done(): return ctx.Err()
               case <-time.After(time.Duration(1<<attempt) * 2 * time.Second):
               }
