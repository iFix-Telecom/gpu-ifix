<objective>
Implement the proactive probe goroutine + refactor the existing `/v1/health/upstreams` handler to report live per-upstream state (closed/half-open/open + last probe result). Probe loop runs every 10s, dispatches 6 synthetic E2E probes in parallel via zero-value `errgroup.Group{}` (critical: NOT `WithContext` per Pitfall 3), shares a 5s deadline, drives `cb.Succeed()`/`Fail()` per D-A4, and batches `UPDATE upstreams SET last_probe_*` writes every 1s (Pitfall 7's WHEN clause in migration 0009 keeps these from triggering reload-storm). The refactored health handler reports the full 6-upstream state derived from `breaker.Set.Snapshot()` + `loader.All()` with a 2s in-memory cache.

Purpose: SC-2 explicitly says "Operator can hit `/v1/health/upstreams` and see live per-upstream state for all six upstreams" and SC-1 ("≤10s failover") relies on the probe loop tripping the breaker before real traffic hits. Without probes, breakers can only open on real-request failures → first client request of a dead upstream pays the latency.

Output: 2 new/refactored files in upstreams/ + 2 test files. Existing health.go is substantially replaced (not appended).

    ```go
    package upstreams

    import (
        "bytes"
        "context"
        _ "embed"
        "encoding/json"
        "fmt"
        "log/slog"
        "mime/multipart"
        "net/http"
        "strconv"
        "sync"
        "time"

        "github.com/jackc/pgx/v5/pgtype"
        "golang.org/x/sync/errgroup"

        "github.com/ifixtelecom/gpu-ifix/gateway/internal/breaker"
        gen "github.com/ifixtelecom/gpu-ifix/gateway/internal/db/gen"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
    )

    //go:embed testdata/probe.wav
    var probeWAV []byte

    // ProbeConfig controls the probe loop cadence + per-probe budget.
    type ProbeConfig struct {
        Interval time.Duration // default 10s (PROBE_INTERVAL_SECONDS)
        Budget   time.Duration // default 5s (PROBE_BUDGET_SECONDS)
    }

    // Probe runs synthetic E2E probes every Interval and mirrors results to
    // the in-process breakers + the upstreams.last_probe_* columns.
    type Probe struct {
        loader  *Loader
        breaker *breaker.Set
        q       *gen.Queries
        cfg     ProbeConfig
        log     *slog.Logger
        client  *http.Client

        updates   chan gen.UpdateUpstreamProbeParams
        dropped   int64
        wg        sync.WaitGroup
    }

    const probeUpdateBufferSize = 100

    // NewProbe constructs the probe loop. Call Run(ctx) in a goroutine at boot.
    func NewProbe(loader *Loader, bs *breaker.Set, q *gen.Queries, cfg ProbeConfig, log *slog.Logger) *Probe {
        if cfg.Interval == 0 {
            cfg.Interval = 10 * time.Second
        }
        if cfg.Budget == 0 {
            cfg.Budget = 5 * time.Second
        }
        return &Probe{
            loader:  loader,
            breaker: bs,
            q:       q,
            cfg:     cfg,
            log:     log.With("module", "PROBE"),
            client:  &http.Client{Timeout: cfg.Budget + 500*time.Millisecond},
            updates: make(chan gen.UpdateUpstreamProbeParams, probeUpdateBufferSize),
        }
    }

    // Run starts the ticker + batch-flush goroutines. Blocks until ctx is canceled.
    func (p *Probe) Run(ctx context.Context) {
        p.wg.Add(1)
        go func() {
            defer p.wg.Done()
            p.flushLoop(ctx)
        }()
        tick := time.NewTicker(p.cfg.Interval)
        defer tick.Stop()
        // Kick off an immediate first probe — don't wait a full Interval at boot.
        p.doTick(ctx)
        for {
            select {
            case <-ctx.Done():
                p.wg.Wait()
                return
            case <-tick.C:
                p.doTick(ctx)
            }
        }
    }

    // doTick dispatches all parallel probes for this cycle. Uses zero-value
    // errgroup.Group{} — one probe's failure MUST NOT cancel siblings (Pitfall 3).
    // Shared 5s deadline comes from context.WithTimeout, NOT errgroup.WithContext.
    func (p *Probe) doTick(parent context.Context) {
        tickCtx, cancel := context.WithTimeout(parent, p.cfg.Budget)
        defer cancel()

        all := p.loader.All()
        breakerSnap := p.breaker.Snapshot()

        // Decide which upstreams to probe this cycle:
        // - tier-0: always
        // - tier-1: only when the same-role tier-0 breaker is OPEN or HALF_OPEN (D-A2)
        tier0ClosedByRole := make(map[string]bool, 3) // role → is tier-0 CLOSED
        for _, u := range all {
            if u.Tier == 0 {
                tier0ClosedByRole[u.Role] = breakerSnap[u.Name] == "closed"
            }
        }

        var g errgroup.Group
        for _, u := range all {
            u := u
            if u.Tier == 1 && tier0ClosedByRole[u.Role] {
                continue // skip external probe on-demand (save OpenRouter/OpenAI cost)
            }
            g.Go(func() error {
                p.probeOne(tickCtx, u)
                return nil // ALWAYS nil to prevent errgroup cascade cancel
            })
        }
        _ = g.Wait()
    }

    // probeOne runs a single synthetic E2E request. Drives the breaker via
    // breaker.Set.Execute so IsSuccessful logic (D-A4) applies uniformly.
    func (p *Probe) probeOne(ctx context.Context, u UpstreamConfig) {
        start := time.Now()
        var status string
        var errMsg string
        _, err := p.breaker.Execute(u.Name, func() (*http.Response, error) {
            return p.dispatch(ctx, u)
        })
        dur := time.Since(start)
        obs.ProbeDurationMs.WithLabelValues(u.Name).Observe(float64(dur.Milliseconds()))
        if err == nil {
            status = "ok"
        } else if ctx.Err() == context.DeadlineExceeded {
            status = "timeout"
            errMsg = "probe budget exceeded"
            obs.ProbeFailureTotal.WithLabelValues(u.Name, "timeout").Inc()
        } else {
            status = "failed"
            errMsg = err.Error()
            obs.ProbeFailureTotal.WithLabelValues(u.Name, "error").Inc()
        }
        p.enqueueUpdate(u.Name, dur, status, errMsg)
    }

    // dispatch builds the role-appropriate synthetic request and issues it.
    func (p *Probe) dispatch(ctx context.Context, u UpstreamConfig) (*http.Response, error) {
        var (
            req *http.Request
            err error
        )
        switch u.Role {
        case "llm":
            body := []byte(`{"model":"qwen","messages":[{"role":"user","content":"ping"}],"max_tokens":1,"temperature":0}`)
            req, err = http.NewRequestWithContext(ctx, http.MethodPost, u.URL+"/v1/chat/completions", bytes.NewReader(body))
            if err != nil { return nil, err }
            req.Header.Set("Content-Type", "application/json")
        case "embed":
            body := []byte(`{"input":"ping","model":"probe-default"}`)
            req, err = http.NewRequestWithContext(ctx, http.MethodPost, u.URL+"/v1/embeddings", bytes.NewReader(body))
            if err != nil { return nil, err }
            req.Header.Set("Content-Type", "application/json")
        case "stt":
            buf := &bytes.Buffer{}
            mw := multipart.NewWriter(buf)
            // model field
            if err := mw.WriteField("model", "whisper-1"); err != nil { return nil, err }
            // file field
            fw, err := mw.CreateFormFile("file", "probe.wav")
            if err != nil { return nil, err }
            if _, err := fw.Write(probeWAV); err != nil { return nil, err }
            if err := mw.Close(); err != nil { return nil, err }
            req, err = http.NewRequestWithContext(ctx, http.MethodPost, u.URL+"/v1/audio/transcriptions", buf)
            if err != nil { return nil, err }
            req.Header.Set("Content-Type", mw.FormDataContentType())
        default:
            return nil, fmt.Errorf("probe: unknown role %q", u.Role)
        }
        if u.AuthBearer != "" {
            req.Header.Set("Authorization", "Bearer "+u.AuthBearer)
        }
        resp, err := p.client.Do(req)
        if err != nil {
            return nil, err
        }
        if resp.StatusCode >= 500 {
            resp.Body.Close()
            return nil, &breaker.HTTPError{Status: resp.StatusCode, Msg: "probe upstream 5xx: " + strconv.Itoa(resp.StatusCode)}
        }
        if resp.StatusCode >= 400 {
            resp.Body.Close()
            // 4xx: IsSuccessful treats as success (client/config issue, not health)
            return nil, &breaker.HTTPError{Status: resp.StatusCode, Msg: "probe 4xx: " + strconv.Itoa(resp.StatusCode)}
        }
        resp.Body.Close()
        return resp, nil
    }

    // enqueueUpdate pushes a DB writeback; non-blocking (drop on overflow with metric).
    func (p *Probe) enqueueUpdate(name string, dur time.Duration, status, errMsg string) {
        params := gen.UpdateUpstreamProbeParams{
            Name:            name,
            LastProbeAt:     pgtype.Timestamptz{Time: time.Now(), Valid: true},
            LastProbeMs:     pgtype.Int4{Int32: int32(dur.Milliseconds()), Valid: true},
            LastProbeStatus: pgtype.Text{String: status, Valid: true},
        }
        if errMsg != "" {
            params.LastProbeError = pgtype.Text{String: errMsg, Valid: true}
        }
        select {
        case p.updates <- params:
        default:
            p.dropped++
        }
    }

    // flushLoop drains up to 100 pending updates every 1s.
    func (p *Probe) flushLoop(ctx context.Context) {
        tick := time.NewTicker(1 * time.Second)
        defer tick.Stop()
        for {
            select {
            case <-ctx.Done():
                p.drain(context.Background()) // best-effort final drain
                return
            case <-tick.C:
                p.drain(ctx)
            }
        }
    }

    func (p *Probe) drain(ctx context.Context) {
        for {
            select {
            case u := <-p.updates:
                dctx, cancel := context.WithTimeout(ctx, 2*time.Second)
                if err := p.q.UpdateUpstreamProbe(dctx, u); err != nil {
                    p.log.Warn("probe UPDATE failed", "upstream", u.Name, "err", err)
                }
                cancel()
            default:
                return
            }
        }
    }

    // Only used by tests; exported via package-level var for test access.
    // Executor may inline or use a helper function if preferred.
    var _ = json.Marshal // keep import if unused above
    ```

    ```go
    func TestProbeLoop_DispatchesToTier0(t *testing.T) {
        ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
        defer cancel()
        var llmHits, sttHits, embedHits atomic.Int64
        mkSrv := func(counter *atomic.Int64) *httptest.Server {
            return httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                counter.Add(1)
                w.WriteHeader(200)
                _, _ = w.Write([]byte(`{"id":"probe"}`))
            }))
        }
        llm := mkSrv(&llmHits); defer llm.Close()
        stt := mkSrv(&sttHits); defer stt.Close()
        embed := mkSrv(&embedHits); defer embed.Close()
        loader, bs, q := setupProbeEnv(t, ctx, llm.URL, stt.URL, embed.URL)
        p := upstreams.NewProbe(loader, bs, q, upstreams.ProbeConfig{
            Interval: 500 * time.Millisecond,
            Budget:   2 * time.Second,
        }, slog.New(slog.NewTextHandler(io.Discard, nil)))
        runCtx, runCancel := context.WithCancel(ctx)
        defer runCancel()
        go p.Run(runCtx)
        // Wait for at least 2 ticks (≈1s)
        time.Sleep(1500 * time.Millisecond)
        if llmHits.Load() < 2 || sttHits.Load() < 2 || embedHits.Load() < 2 {
            t.Fatalf("tier-0 hits: llm=%d stt=%d embed=%d (want ≥2 each)", llmHits.Load(), sttHits.Load(), embedHits.Load())
        }
    }

    func TestProbeLoop_OneFailureDoesNotCancelSiblings(t *testing.T) {
        ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
        defer cancel()
        var llmHits, sttHits atomic.Int64
        llm := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            llmHits.Add(1)
            w.WriteHeader(500) // fast failure
        }))
        defer llm.Close()
        stt := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            sttHits.Add(1)
            time.Sleep(1 * time.Second) // slow but within budget
            w.WriteHeader(200)
        }))
        defer stt.Close()
        embed := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            w.WriteHeader(200)

        }
    }
    ```

    ```go
    // Package upstreams aggregates per-upstream runtime state and exposes it
    // via GET /v1/health/upstreams. Phase 3 replaces the Phase 1 health-bridge
    // aggregator with an in-process view over breaker.Set.Snapshot and
    // Loader.All. The pod's :9100 aggregator remains available for internal
    // debug but is no longer the authority.
    package upstreams

    import (
        "encoding/json"
        "log/slog"
        "net/http"
        "sync"
        "time"

        "github.com/ifixtelecom/gpu-ifix/gateway/internal/breaker"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/httpx"
    )

    const healthCacheTTL = 2 * time.Second

    type healthResponse struct {
        Status    string                    `json:"status"`
        Upstreams map[string]upstreamStatus `json:"upstreams"`
    }

    type upstreamStatus struct {
        State           string `json:"state"` // "closed"|"half-open"|"open"
        Role            string `json:"role"`
        Tier            int    `json:"tier"`
        LastProbeMs     *int   `json:"last_probe_ms,omitempty"`
        LastProbeAt     string `json:"last_probe_at,omitempty"`
        LastProbeStatus string `json:"last_probe_status,omitempty"`
    }

    type cachedHealthResponse struct {
        storedAt time.Time
        body     []byte
        status   int
    }

    // NewHealthHandler returns the HTTP handler for GET /v1/health/upstreams.
    // CONTEXT.md D-D1 Plumbing — derives status from breaker state + loader config.
    func NewHealthHandler(loader *Loader, bs *breaker.Set, log *slog.Logger) http.HandlerFunc {
        log = log.With("module", "UPSTREAMS_HEALTH")
        var (
            mu    sync.Mutex
            cache cachedHealthResponse
        )
        return func(w http.ResponseWriter, r *http.Request) {
            mu.Lock()
            if time.Since(cache.storedAt) < healthCacheTTL && cache.body != nil {
                b, s := cache.body, cache.status
                mu.Unlock()
                w.Header().Set("Content-Type", "application/json")
                w.WriteHeader(s)
                _, _ = w.Write(b)
                return
            }
            mu.Unlock()

            all := loader.All()
            snap := bs.Snapshot()

            resp := healthResponse{
                Upstreams: make(map[string]upstreamStatus, len(all)),
            }

            // Track per-role: does ANY tier have a CLOSED breaker? And specifically is tier-0 CLOSED?
            tier0Closed := make(map[string]bool, 3)
            roleHasClosed := make(map[string]bool, 3)
            for _, u := range all {
                st, ok := snap[u.Name]
                if !ok {
                    st = "unknown"
                }
                us := upstreamStatus{
                    State: st,
                    Role:  u.Role,
                    Tier:  u.Tier,
                }
                resp.Upstreams[u.Name] = us
                if st == "closed" {
                    roleHasClosed[u.Role] = true
                    if u.Tier == 0 {
                        tier0Closed[u.Role] = true
                    }
                }
            }

            // Derive top-level status
            httpStatus := http.StatusOK
            switch {
            case allTier0Closed(all, tier0Closed):
                resp.Status = "ok"
            case allRolesHaveAnyClosed(all, roleHasClosed):
                resp.Status = "degraded"
            default:
                resp.Status = "failed"
                httpStatus = http.StatusServiceUnavailable
            }

            body, err := json.Marshal(resp)
            if err != nil {
                log.Error("marshal health response", "err", err, "request_id", httpx.RequestIDFrom(r.Context()))
                httpx.WriteOpenAIError(w, http.StatusInternalServerError, "api_error", "health_encoding_failed", "Failed to encode health response.")
                return
            }

            mu.Lock()
            cache = cachedHealthResponse{
                storedAt: time.Now(),
                body:     body,
                status:   httpStatus,
            }
            mu.Unlock()

            w.Header().Set("Content-Type", "application/json")
            w.WriteHeader(httpStatus)
            _, _ = w.Write(body)
        }
    }

    // allTier0Closed reports whether every role represented in the upstream
    // list has its tier-0 upstream in CLOSED state.
    func allTier0Closed(all []UpstreamConfig, tier0Closed map[string]bool) bool {
        seen := make(map[string]bool, 3)
        for _, u := range all {
            if u.Tier == 0 {
                seen[u.Role] = true
            }
        }
        for role := range seen {
            if !tier0Closed[role] {
                return false
            }
        }
        return len(seen) > 0
    }

    // allRolesHaveAnyClosed reports whether every role has at least one closed
    // upstream across all tiers.
    func allRolesHaveAnyClosed(all []UpstreamConfig, roleHasClosed map[string]bool) bool {
        seen := make(map[string]bool, 3)
        for _, u := range all {
            seen[u.Role] = true
        }
        for role := range seen {
            if !roleHasClosed[role] {
                return false
            }
        }
        return len(seen) > 0
    }
    ```
