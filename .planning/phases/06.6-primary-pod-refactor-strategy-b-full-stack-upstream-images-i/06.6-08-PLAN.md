func (r *Reconciler) SubscribePrimaryEvents(ctx context.Context) {
    pubsub := redisx.SubscribePrimaryEvents(ctx, r.deps.Redis)
    defer pubsub.Close()
    for msg := range pubsub.Channel() {
        var ev redisx.PrimaryEvent
        if err := json.Unmarshal([]byte(msg.Payload), &ev); err != nil {
            continue
        }
        if !r.isLeader.Load() {
            continue  // non-leader replicas drop
        }
        switch ev.Type {
        case "primary_ready":
            if r.deps.FSM.State() == StateEmergencyActive {
                r.cancelActiveLifecycle(ctx, "primary_took_over")
            }
        // future: more inter-reconciler signals
        }
    }
}
