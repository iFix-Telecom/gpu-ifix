INSERT INTO ai_gateway.primary_lifecycles (trigger_reason, leader_replica)

UPDATE ai_gateway.primary_lifecycles
SET first_health_pass_at = NOW(),
    events = events || $2::jsonb
WHERE id = $1;

-- name: MarkPrimaryLifecycleDraining :exec     -- NEW vs emerg
UPDATE ai_gateway.primary_lifecycles
SET drain_started_at = NOW(),
    events = events || $2::jsonb
WHERE id = $1;

-- name: ClosePrimaryLifecycle :exec
UPDATE ai_gateway.primary_lifecycles
SET ended_at = NOW(),
    shutdown_reason = $2, total_cost_brl = $3,
    events = events || $4::jsonb
WHERE id = $1 AND ended_at IS NULL;

package redisx

const (
    PrimaryEventsChannel  = "gw:primary:events"
    primaryStateKeyPrefix = "gw:primary:"
    primaryLockKey        = "gw:primary:lock"
)

func PrimaryStateKey() string { return primaryStateKeyPrefix + "state" }
func PrimaryLockKey() string  { return primaryLockKey }

type PrimaryEvent struct {
    Type        string `json:"type"`        // "schedule_up_fired" | "provisioning_started" | "primary_ready" | "draining_started" | "destroyed" | "force_up_request" | "force_down_request" | "cancel_in_flight"
    State       string `json:"state"`
    LifecycleID int64  `json:"lifecycle_id"`
    Reason      string `json:"reason"`
    SinceUnix   int64  `json:"since_unix"`
    ReplicaID   string `json:"replica_id"`
}

func WritePrimaryState(ctx, rdb, ...) error              // analog WriteEmergState
func PublishPrimaryEvent(ctx, rdb, ev PrimaryEvent) error  // analog PublishEmergEvent
func SubscribePrimaryEvents(ctx, rdb) *redis.PubSub        // analog SubscribeEmergEvents
func NewPrimaryRedsync(rdb) *redsync.Redsync               // analog NewEmergRedsync (separate lock key)
