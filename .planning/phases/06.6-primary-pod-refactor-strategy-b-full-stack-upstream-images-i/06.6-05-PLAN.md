```go
// ScheduleRule is constructed once at boot from cfg fields; immutable thereafter.
type ScheduleRule struct {
    Timezone        *time.Location // resolved via time.LoadLocation(cfg.PrimaryPodScheduleTimezone)
    UpHour          int
    DownHour        int
    Days            map[time.Weekday]bool // populated from CSV "mon,tue,wed,thu,fri"
    GraceRampDownS  int
    Disabled        bool
}

func ParseScheduleEnv(cfg config.Config) (ScheduleRule, error)  // fail-fast on tz error
func (r ScheduleRule) IsInPeak(now time.Time) bool              // handles overnight + days + disabled
func (r ScheduleRule) NextTransition(now time.Time) (time.Time, string)  // ("up"|"down")
```

```go
type State int32

const (
    StateAsleep State = iota
    StateProvisioning
    StateReady
    StateDraining
    StateDestroying
)

var allStates = []State{StateAsleep, StateProvisioning, StateReady, StateDraining, StateDestroying}

// stateChangeWriter is the audit-decoupling interface (mirrors emerg/fsm.go:58-60)
type stateChangeWriter interface {
    WriteStateChange(kind string, ev /*audit.Event*/ any) error  // shape from emerg.audit.Writer
}

type FSM struct {
    state       atomic.Int32              // holds State
    onChange    func(from, to State, at time.Time, reason string)  // post-CAS callback
    writer      stateChangeWriter         // optional, may be nil in tests
    enteredAt   atomic.Int64              // unix nanos of last transition
}

func NewFSM(writer stateChangeWriter, onChange func(from, to State, at time.Time, reason string)) *FSM
func (f *FSM) State() State                                       // lockless read
func (f *FSM) Transition(from, to State, at time.Time, reason string) error  // CAS-guarded
func (f *FSM) SetState(to State, at time.Time, reason string)     // unconditional set (recovery path; mirror emerg SetState)
func (f *FSM) EnteredAt() time.Time
```
