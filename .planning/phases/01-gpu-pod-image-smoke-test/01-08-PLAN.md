1. `prepare` — validates inputs, fetches ghcr.io tag existence (to fail fast if image not built)
2. `create-pod` — searches offers, accepts cheapest under $max, creates instance, waits for `actual_status==running` + SSH reachable (timeout 15 min)
3. `provision` — SCPs docker-compose.yml + .env + onstart.sh + scripts/ to pod, invokes onstart, waits for /health/ready != unknown (timeout 10 min)

5. `archive` — uploads report as workflow artifact; on success, posts a build summary
6. `destroy` (always) — `DELETE /api/v0/instances/{id}/` even if prior jobs failed

- 1 → workflow failure, smoke.py crashed (investigation required)
- 2..5 → workflow failure, specific gate failed (reported in summary)
- 6 → workflow failure, multiple gates failed

Cost guardrails (D-22):
- `max_price_per_hour` input — cannot accept an offer above this
- workflow timeout 45 min → force-kill → destroy step
- create-pod retries up to 3 different offers on initial failure; if all 3 fail, abort and alert

Create `pod/scripts/vast-ai.sh` — a small imperative-shell CLI that smoke.yml invokes. Keeps REST logic out of the YAML where it's hard to test. Usage:

```
pod/scripts/vast-ai.sh search [--max-price N] [--gpu "RTX 4090"]
pod/scripts/vast-ai.sh create --offer-id N --image TAG --env-file PATH --onstart PATH
pod/scripts/vast-ai.sh status --instance-id N
pod/scripts/vast-ai.sh wait-running --instance-id N --timeout-seconds N
pod/scripts/vast-ai.sh destroy --instance-id N
pod/scripts/vast-ai.sh ssh-exec --instance-id N --cmd "..."
pod/scripts/vast-ai.sh scp-upload --instance-id N --src LOCAL --dest REMOTE
```

Requires `VAST_AI_API_KEY` in env. Writes structured output to stdout (one JSON-encoded event per line for GHA parsing).

```bash
#!/usr/bin/env bash
# pod/scripts/vast-ai.sh — Bash wrapper around Vast.ai REST API.
# Used by .github/workflows/smoke.yml (D-22) to orchestrate ephemeral pods.

set -euo pipefail

: "${VAST_AI_API_KEY:?missing}"
VAST_BASE="${VAST_BASE:-https://vast.ai/api/v0}"

log() { printf '[%s] [vast-ai] %s\n' "$(date -Iseconds)" "$*" >&2; }

# --- subcommand dispatch -------------------------------------------------
cmd="${1:-}"; shift || true

api() {
  # api METHOD PATH [JSON_BODY]
  local method="$1" path="$2" body="${3:-}"
  local url="${VAST_BASE}${path}"
  local args=(-sS -X "${method}" -H "Authorization: Bearer ${VAST_AI_API_KEY}")
  if [[ -n "${body}" ]]; then
    args+=(-H "Content-Type: application/json" --data-raw "${body}")
  fi
  curl "${args[@]}" "${url}"
}

case "${cmd}" in

  search)
    # search [--max-price N] [--gpu "RTX 4090"]
    MAX_PRICE="0.40"
    GPU="RTX 4090"
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --max-price) MAX_PRICE="$2"; shift 2;;
        --gpu) GPU="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    # Vast.ai search accepts a "q" JSON filter.
    Q=$(jq -cn --arg gpu "$GPU" --argjson max "$MAX_PRICE" '{
      gpu_name:$gpu, num_gpus:1, disk_space:{gte:50},
      reliability:{gte:0.98}, inet_down:{gte:200},
      rentable:true, verified:{eq:true},
      dph_total:{lte:$max},
      order:[["dph_total","asc"]], limit:5
    }')
    api GET "/bundles/?q=${Q}"
    ;;

  create)
    # create --offer-id N --image TAG --env-file PATH --onstart PATH
    OFFER=""; IMAGE=""; ENV_FILE=""; ONSTART=""
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --offer-id)  OFFER="$2"; shift 2;;
        --image)     IMAGE="$2"; shift 2;;
        --env-file)  ENV_FILE="$2"; shift 2;;
        --onstart)   ONSTART="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    : "${OFFER:?}" "${IMAGE:?}" "${ENV_FILE:?}" "${ONSTART:?}"
    # Build env map from env-file (KEY=VALUE, one per line, comments allowed)
    ENV_JSON=$(awk -F= '!/^#/ && NF>=2 {gsub(/"/,"\\\"",$2); printf "\"%s\":\"%s\",", $1, $2}' "$ENV_FILE" \
               | sed 's/,$//' | awk '{print "{" $0 "}"}')
    ONSTART_B64=$(base64 -w0 "$ONSTART")
    BODY=$(jq -cn --arg img "$IMAGE" --argjson env "$ENV_JSON" --arg onstart_b64 "$ONSTART_B64" '{
      client_id:"me",
      image:$img,
      env:$env,
      onstart:"echo \($onstart_b64) | base64 -d > /root/onstart.sh && chmod +x /root/onstart.sh && /root/onstart.sh",
      runtype:"ssh",
      disk:60,
      label:"ifix-smoke"
    }')
    api PUT "/asks/${OFFER}/" "${BODY}"
    ;;

  status)
    # status --instance-id N
    ID=""
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --instance-id) ID="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    : "${ID:?}"
    api GET "/instances/${ID}/"
    ;;

  wait-running)
    # wait-running --instance-id N --timeout-seconds N
    ID=""; TIMEOUT=900
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --instance-id)      ID="$2"; shift 2;;
        --timeout-seconds)  TIMEOUT="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    : "${ID:?}"
    DEADLINE=$(( $(date +%s) + TIMEOUT ))
    while [[ $(date +%s) -lt ${DEADLINE} ]]; do
      S=$(api GET "/instances/${ID}/" || true)
      ACTUAL=$(printf '%s' "$S" | jq -r '.instances.actual_status // .instances[0].actual_status // empty')
      SSH_HOST=$(printf '%s' "$S" | jq -r '.instances.ssh_host // .instances[0].ssh_host // empty')
      SSH_PORT=$(printf '%s' "$S" | jq -r '.instances.ssh_port // .instances[0].ssh_port // empty')
      IP=$(printf '%s' "$S" | jq -r '.instances.public_ipaddr // .instances[0].public_ipaddr // empty')
      log "instance ${ID}: actual_status=${ACTUAL}  ip=${IP}  ssh=${SSH_HOST}:${SSH_PORT}"
      if [[ "${ACTUAL}" == "running" && -n "${SSH_HOST}" && -n "${SSH_PORT}" ]]; then
        printf '%s' "$S"
        exit 0
      fi
      sleep 15
    done
    log "timeout waiting for instance ${ID} to reach running"
    exit 1
    ;;

  destroy)
    ID=""
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --instance-id) ID="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    : "${ID:?}"
    log "destroying instance ${ID}"
    api DELETE "/instances/${ID}/"
    ;;

  ssh-exec)
    # ssh-exec --ssh-host HOST --ssh-port PORT --cmd "..."
    SSH_HOST=""; SSH_PORT=""; CMD=""
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --ssh-host) SSH_HOST="$2"; shift 2;;
        --ssh-port) SSH_PORT="$2"; shift 2;;
        --cmd)      CMD="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    : "${SSH_HOST:?}" "${SSH_PORT:?}" "${CMD:?}"
    ssh -o StrictHostKeyChecking=accept-new -o UserKnownHostsFile=/tmp/known_hosts \
        -p "${SSH_PORT}" "root@${SSH_HOST}" "${CMD}"
    ;;

  scp-upload)
    # scp-upload --ssh-host HOST --ssh-port PORT --src LOCAL --dest REMOTE
    SSH_HOST=""; SSH_PORT=""; SRC=""; DEST=""
    while [[ $# -gt 0 ]]; do
      case "$1" in
        --ssh-host) SSH_HOST="$2"; shift 2;;
        --ssh-port) SSH_PORT="$2"; shift 2;;
        --src)      SRC="$2"; shift 2;;
        --dest)     DEST="$2"; shift 2;;
        *) log "unknown arg $1"; exit 2;;
      esac
    done
    : "${SSH_HOST:?}" "${SSH_PORT:?}" "${SRC:?}" "${DEST:?}"
    scp -o StrictHostKeyChecking=accept-new -o UserKnownHostsFile=/tmp/known_hosts \
        -P "${SSH_PORT}" -r "${SRC}" "root@${SSH_HOST}:${DEST}"
    ;;

  *)
    cat >&2 <<EOF
usage:
  $0 search [--max-price N] [--gpu GPU_NAME]
  $0 create --offer-id N --image TAG --env-file PATH --onstart PATH
  $0 status --instance-id N
  $0 wait-running --instance-id N [--timeout-seconds N]
  $0 destroy --instance-id N
  $0 ssh-exec --ssh-host HOST --ssh-port PORT --cmd "..."
  $0 scp-upload --ssh-host HOST --ssh-port PORT --src L --dest R

env: VAST_AI_API_KEY (required), VAST_BASE (optional, default https://vast.ai/api/v0)
EOF
    exit 2
    ;;
esac
```

  workflow_dispatch:

        type: string
      weights_whisper_key:
        description: 'MinIO object key for Whisper tarball'

        description: 'Hard workflow timeout (D-22 cost cap)'
        required: false
        default: '45'
        type: string

# D-22: smoke runs must NOT be cancelled mid-flight — cost risk of leaked pod.
concurrency:
  group: smoke-${{ github.ref }}
  cancel-in-progress: false

permissions:
  contents: read

env:
  IMAGE_POD_REPO: ghcr.io/ifixtelecom/ifix-ai-pod

jobs:
  smoke:
    name: Smoke-test on Vast.ai 4090
    runs-on: ubuntu-latest
    timeout-minutes: ${{ fromJSON(inputs.smoke_timeout_minutes) }}
    env:
      VAST_AI_API_KEY:    ${{ secrets.VAST_AI_API_KEY }}
      MINIO_ENDPOINT:     ${{ secrets.MINIO_ENDPOINT }}
      MINIO_ACCESS_KEY:   ${{ secrets.MINIO_ACCESS_KEY }}
      MINIO_SECRET_KEY:   ${{ secrets.MINIO_SECRET_KEY }}
      MINIO_BUCKET:       ${{ secrets.MINIO_BUCKET }}
      WEIGHTS_QWEN_SHA256:    ${{ secrets.WEIGHTS_QWEN_SHA256 }}
      WEIGHTS_WHISPER_SHA256: ${{ secrets.WEIGHTS_WHISPER_SHA256 }}
      WEIGHTS_BGE_M3_SHA256:  ${{ secrets.WEIGHTS_BGE_M3_SHA256 }}
      IMAGE_TAG:          ${{ inputs.image_tag }}
      HB_TAG:             ${{ inputs.health_bridge_tag != '' && inputs.health_bridge_tag || inputs.image_tag }}
      WEIGHTS_QWEN_KEY:   ${{ inputs.weights_qwen_key }}
      WEIGHTS_WHISPER_KEY: ${{ inputs.weights_whisper_key }}
      WEIGHTS_BGE_M3_KEY: ${{ inputs.weights_bge_m3_key }}
      MAX_PRICE:          ${{ inputs.max_price_per_hour }}

          set -euo pipefail
          echo "=== Preparing smoke run ==="
          for s in VAST_AI_API_KEY MINIO_ENDPOINT MINIO_ACCESS_KEY MINIO_SECRET_KEY MINIO_BUCKET \
                   WEIGHTS_QWEN_SHA256 WEIGHTS_WHISPER_SHA256 WEIGHTS_BGE_M3_SHA256; do

      - name: Setup Python + jq
        run: |
          sudo apt-get update && sudo apt-get install -y jq openssh-client
          python3 --version

      - name: Install Python deps

          python3 -m pip install --upgrade pip

          OFFERS_JSON=$(pod/scripts/vast-ai.sh search --max-price "${MAX_PRICE}")
          OFFER_ID=$(echo "$OFFERS_JSON" | jq -r '.offers[0].id // .bundles[0].id // empty')
          PRICE=$(echo "$OFFERS_JSON" | jq -r '.offers[0].dph_total // .bundles[0].dph_total // empty')
          if [[ -z "$OFFER_ID" ]]; then
            echo "::error::no Vast.ai offer matched filters under \$${MAX_PRICE}/h"
            echo "offers body:"; echo "$OFFERS_JSON"
            exit 1
          fi
          echo "chose offer ${OFFER_ID} at \$${PRICE}/h"
          echo "offer_id=${OFFER_ID}" >> "$GITHUB_OUTPUT"
          echo "price=${PRICE}"     >> "$GITHUB_OUTPUT"

      - name: Materialize env-file for create
        run: |
          set -euo pipefail
          umask 077
          cat > /tmp/pod.env <<EOF
IFIX_AI_POD_IMAGE=${IMAGE_POD_REPO}:${IMAGE_TAG}
IFIX_AI_POD_HEALTH_BRIDGE_IMAGE=ghcr.io/ifixtelecom/ifix-ai-pod-health-bridge:${HB_TAG}
WEIGHTS_DIR=/weights
MINIO_ENDPOINT=${MINIO_ENDPOINT}
MINIO_BUCKET=${MINIO_BUCKET}
MINIO_ACCESS_KEY=${MINIO_ACCESS_KEY}
MINIO_SECRET_KEY=${MINIO_SECRET_KEY}
WEIGHTS_QWEN_KEY=${WEIGHTS_QWEN_KEY}
WEIGHTS_WHISPER_KEY=${WEIGHTS_WHISPER_KEY}
WEIGHTS_BGE_M3_KEY=${WEIGHTS_BGE_M3_KEY}
WEIGHTS_QWEN_SHA256=${WEIGHTS_QWEN_SHA256}
WEIGHTS_WHISPER_SHA256=${WEIGHTS_WHISPER_SHA256}
WEIGHTS_BGE_M3_SHA256=${WEIGHTS_BGE_M3_SHA256}
ENV=production
LOG_LEVEL=info
EOF

      - name: Create Vast.ai instance
        id: create
        run: |
          set -euo pipefail
          RESP=$(pod/scripts/vast-ai.sh create \
            --offer-id "${{ steps.search.outputs.offer_id }}" \
            --image "${IMAGE_POD_REPO}:${IMAGE_TAG}" \
            --env-file /tmp/pod.env \
            --onstart pod/onstart.sh)
          echo "create response:"; echo "$RESP"
          INST_ID=$(echo "$RESP" | jq -r '.new_contract // .instance_id // .id // empty')
          if [[ -z "$INST_ID" ]]; then
            echo "::error::could not parse instance id from create response"
            exit 1
          fi
          echo "instance_id=${INST_ID}" >> "$GITHUB_OUTPUT"
          echo "started_at=$(date +%s)"  >> "$GITHUB_OUTPUT"

      - name: Wait for instance running + SSH ready
        id: ready
        run: |
          set -euo pipefail
          STATE=$(pod/scripts/vast-ai.sh wait-running \
            --instance-id "${{ steps.create.outputs.instance_id }}" \
            --timeout-seconds 900)
          echo "$STATE" > /tmp/instance.json
          SSH_HOST=$(jq -r '.instances.ssh_host // .instances[0].ssh_host' /tmp/instance.json)
          SSH_PORT=$(jq -r '.instances.ssh_port // .instances[0].ssh_port' /tmp/instance.json)
          PUB_IP=$(jq -r   '.instances.public_ipaddr // .instances[0].public_ipaddr' /tmp/instance.json)
          echo "ssh_host=${SSH_HOST}"   >> "$GITHUB_OUTPUT"
          echo "ssh_port=${SSH_PORT}"   >> "$GITHUB_OUTPUT"
          echo "public_ip=${PUB_IP}"    >> "$GITHUB_OUTPUT"

      - name: Upload compose + scripts to pod
        run: |
          set -euo pipefail
          echo "=== Provisioning pod ==="
          HOST="${{ steps.ready.outputs.ssh_host }}"
          PORT="${{ steps.ready.outputs.ssh_port }}"

          # Stage files into /tmp/pod-bundle/ locally, then scp once.
          rm -rf /tmp/pod-bundle && mkdir -p /tmp/pod-bundle/scripts
          cp pod/docker-compose.yml /tmp/pod-bundle/
          cp /tmp/pod.env           /tmp/pod-bundle/.env
          cp pod/onstart.sh         /tmp/pod-bundle/
          cp pod/scripts/download-weights.sh /tmp/pod-bundle/scripts/
          chmod +x /tmp/pod-bundle/onstart.sh /tmp/pod-bundle/scripts/download-weights.sh

          # Ensure target dir on pod exists
          pod/scripts/vast-ai.sh ssh-exec --ssh-host "$HOST" --ssh-port "$PORT" \
            --cmd "mkdir -p /opt/ifix-ai-pod/scripts"
          pod/scripts/vast-ai.sh scp-upload --ssh-host "$HOST" --ssh-port "$PORT" \
            --src "/tmp/pod-bundle/." \
            --dest "/opt/ifix-ai-pod/"

          # Note: onstart already ran once at pod creation via the create body's `onstart` param.
          # If the pod was booted before our files landed (race), re-invoke onstart manually.
          pod/scripts/vast-ai.sh ssh-exec --ssh-host "$HOST" --ssh-port "$PORT" \
            --cmd "cd /opt/ifix-ai-pod && IFIX_AI_POD_ROOT=/opt/ifix-ai-pod COMPOSE_FILE=/opt/ifix-ai-pod/docker-compose.yml ENV_FILE=/opt/ifix-ai-pod/.env /opt/ifix-ai-pod/onstart.sh 2>&1 | tail -200"

      - name: Run smoke-test (plan 06)
        id: smoke
        run: |
          set +e
          echo "=== Running smoke-test ==="
          TARGET="http://${{ steps.ready.outputs.public_ip }}"
          python3 pod/smoke/smoke.py --target "$TARGET" --out smoke-report.json
          CODE=$?
          set -e
          echo "smoke.py exit code: ${CODE}"
          echo "exit_code=${CODE}" >> "$GITHUB_OUTPUT"
          if [[ -f smoke-report.json ]]; then
            jq . smoke-report.json
          fi
          # Bubble the exit code at a later step so archive + destroy still run.

      - name: Archive smoke-report.json
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: smoke-report-${{ inputs.image_tag }}-${{ github.run_id }}
          path: smoke-report.json
          retention-days: 90
          if-no-files-found: warn

      - name: Publish summary
        if: always()
        env:
          EXIT: ${{ steps.smoke.outputs.exit_code }}
          OFFER_ID: ${{ steps.search.outputs.offer_id }}
          PRICE: ${{ steps.search.outputs.price }}
          INST_ID: ${{ steps.create.outputs.instance_id }}
        run: |
          {
            echo "## Smoke Summary"
            echo ""
            echo "| Field | Value |"
            echo "|---|---|"
            echo "| Image tag | \`${IMAGE_POD_REPO}:${IMAGE_TAG}\` |"
            echo "| Offer | #${OFFER_ID} @ \$${PRICE}/h |"
            echo "| Instance | #${INST_ID} |"
            echo "| smoke.py exit | ${EXIT:-unset} |"
            echo ""
            if [[ -f smoke-report.json ]]; then
              echo "### Gates (D-19)"
              echo '```json'
              jq '.gates' smoke-report.json
              echo '```'
              echo "### Key metrics"
              echo '```json'
              jq '{vram_peak_gb, vram_p95_gb, llm_p95_ttft_ms, llm_p95_tpot_ms, whisper_latency_s, embed_p95_ms, tool_call_valid, errors}' smoke-report.json
              echo '```'
            else
              echo "_smoke-report.json not produced_"
            fi
          } >> "$GITHUB_STEP_SUMMARY"

      - name: Destroy Vast.ai instance (ALWAYS — D-22 cost cap)
        if: always() && steps.create.outputs.instance_id != ''
        run: |
          set -euo pipefail
          echo "=== Destroying pod ==="
          pod/scripts/vast-ai.sh destroy --instance-id "${{ steps.create.outputs.instance_id }}"

      - name: Fail workflow on smoke exit code
        if: always()
        run: |
          CODE="${{ steps.smoke.outputs.exit_code }}"
          if [[ -z "$CODE" ]]; then
            echo "::error::smoke step did not emit exit_code (earlier step failed?)"
            exit 1
          fi
          if [[ "$CODE" != "0" ]]; then
            echo "::error::smoke.py exit ${CODE} — D-19 gates failed. See Summary + smoke-report.json artifact."
            exit "$CODE"
          fi
          echo "D-19 gates GREEN — image ${IMAGE_POD_REPO}:${IMAGE_TAG} is a stable-tag candidate (D-23)."
