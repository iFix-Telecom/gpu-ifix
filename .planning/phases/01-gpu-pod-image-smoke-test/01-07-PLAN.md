```yaml
name: build-pod

on:
  # IMPORTANT: do NOT add `paths:` under `push:` — GitHub applies it to ALL push
  # events (branches + tags). D-23 stable promotion is `git tag v1.0.0` on an
  # already-smoked SHA; that tag push rarely modifies pod/** in the tag commit
  # itself, so a paths filter here would silently block stable releases.
  # Keep paths filter on pull_request only.
  push:
    branches: [main, develop]
    tags: ['v*']
  pull_request:
    branches: [main, develop]
    paths:
      - 'pod/**'
      - 'pkg/openai/**'
      - 'go.mod'
      - 'go.sum'
      - '.github/workflows/build-pod.yml'
  workflow_dispatch:
    inputs:
      tag:
        description: 'Optional explicit tag (else branch-sha)'
        required: false
        type: string

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

permissions:
  contents: read
  packages: write

env:
  REGISTRY: ghcr.io
  IMAGE_POD: ghcr.io/ifixtelecom/ifix-ai-pod
  IMAGE_HEALTH_BRIDGE: ghcr.io/ifixtelecom/ifix-ai-pod-health-bridge
  GO_VERSION: '1.23'

jobs:
  test:
    name: Go unit tests + template integrity
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Go
        uses: actions/setup-go@v5
        with:
          go-version: ${{ env.GO_VERSION }}
          cache: true

      - name: Verify template SHA-256 (plan 02)
        run: |
          set -euo pipefail
          EXPECTED="$(tr -d '[:space:]' < pod/templates/qwen3.5-27b-tool-calling.jinja.sha256)"
          ACTUAL="$(sha256sum pod/templates/qwen3.5-27b-tool-calling.jinja | awk '{print $1}')"
          if [[ "${EXPECTED}" != "${ACTUAL}" ]]; then
            echo "::error::Qwen template SHA-256 drift — expected ${EXPECTED} got ${ACTUAL}"
            exit 1
          fi
          echo "template integrity OK (${ACTUAL:0:12}...)"

      - name: go vet
        run: go vet ./...

      - name: go test
        run: go test ./... -count=1 -race -timeout=3m

      - name: gofmt check
        run: |
          UNFORMATTED="$(gofmt -l .)"
          if [[ -n "${UNFORMATTED}" ]]; then
            echo "::error::gofmt unclean files:"
            echo "${UNFORMATTED}"
            exit 1
          fi

  compute-tags:
    name: Compute image tags
    runs-on: ubuntu-latest
    needs: [test]
    outputs:
      pod_tags:           ${{ steps.set.outputs.pod_tags }}
      health_bridge_tags: ${{ steps.set.outputs.health_bridge_tags }}
      is_stable_release:  ${{ steps.set.outputs.is_stable_release }}
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Compute tags (D-21 + D-23)
        id: set
        env:
          REF:  ${{ github.ref }}
          SHA:  ${{ github.sha }}
          INPUT_TAG: ${{ inputs.tag }}
        run: |
          set -euo pipefail
          SHORT_SHA="$(echo "${SHA}" | cut -c1-7)"
          POD_TAGS=""
          HB_TAGS=""
          IS_STABLE="false"

          if [[ "${REF}" == refs/tags/v* ]]; then
            # Stable release promotion (D-23) — git tag push
            VER="${REF##refs/tags/}"
            POD_TAGS="${IMAGE_POD}:${VER}
          ${IMAGE_POD}:latest
          ${IMAGE_POD}:${VER}-${SHORT_SHA}"
            HB_TAGS="${IMAGE_HEALTH_BRIDGE}:${VER}
          ${IMAGE_HEALTH_BRIDGE}:latest
          ${IMAGE_HEALTH_BRIDGE}:${VER}-${SHORT_SHA}"
            IS_STABLE="true"
          elif [[ "${REF}" == refs/heads/main ]]; then
            POD_TAGS="${IMAGE_POD}:main
          ${IMAGE_POD}:main-${SHORT_SHA}"
            HB_TAGS="${IMAGE_HEALTH_BRIDGE}:main
          ${IMAGE_HEALTH_BRIDGE}:main-${SHORT_SHA}"
          elif [[ "${REF}" == refs/heads/develop ]]; then
            POD_TAGS="${IMAGE_POD}:develop
          ${IMAGE_POD}:develop-${SHORT_SHA}
          ${IMAGE_POD}:latest-dev"
            HB_TAGS="${IMAGE_HEALTH_BRIDGE}:develop
          ${IMAGE_HEALTH_BRIDGE}:develop-${SHORT_SHA}
          ${IMAGE_HEALTH_BRIDGE}:latest-dev"
          elif [[ -n "${INPUT_TAG}" ]]; then
            POD_TAGS="${IMAGE_POD}:${INPUT_TAG}"
            HB_TAGS="${IMAGE_HEALTH_BRIDGE}:${INPUT_TAG}"
          else
            # PR or other — ephemeral sha tag
            POD_TAGS="${IMAGE_POD}:pr-${SHORT_SHA}"
            HB_TAGS="${IMAGE_HEALTH_BRIDGE}:pr-${SHORT_SHA}"
          fi

          # Use heredoc WITHOUT leading whitespace on tag lines (they are the
          # literal tag values going to GITHUB_OUTPUT). Using `printf` for scalars
          # and a tightly-packed heredoc for multi-line outputs avoids the risk
          # of a stray leading space on a tag value that would break pushes.
          {
            echo "pod_tags<<EOF"
            printf '%s\n' "${POD_TAGS}"
            echo "EOF"
            echo "health_bridge_tags<<EOF"
            printf '%s\n' "${HB_TAGS}"
            echo "EOF"
            printf 'is_stable_release=%s\n' "${IS_STABLE}"
          } >> "$GITHUB_OUTPUT"

          echo "resolved tags:"
          echo "pod:"; echo "${POD_TAGS}"
          echo "health-bridge:"; echo "${HB_TAGS}"

  build-pod:
    name: Build & push pod image
    runs-on: ubuntu-latest
    needs: [test, compute-tags]
    timeout-minutes: 60
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Buildx
        uses: docker/setup-buildx-action@v3

      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build & push pod
        uses: docker/build-push-action@v6
        with:
          context: .
          file: pod/Dockerfile
          push: true
          platforms: linux/amd64
          tags: ${{ needs.compute-tags.outputs.pod_tags }}
          cache-from: type=gha,scope=ifix-ai-pod
          cache-to:   type=gha,scope=ifix-ai-pod,mode=max
          labels: |
            org.opencontainers.image.source=${{ github.server_url }}/${{ github.repository }}
            org.opencontainers.image.revision=${{ github.sha }}
            org.opencontainers.image.created=${{ github.event.head_commit.timestamp }}

  build-health-bridge:
    name: Build & push health-bridge image
    runs-on: ubuntu-latest
    needs: [test, compute-tags]
    timeout-minutes: 15
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Buildx
        uses: docker/setup-buildx-action@v3

      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build & push health-bridge
        uses: docker/build-push-action@v6
        with:
          context: .
          file: pod/health-bridge/Dockerfile
          push: true
          platforms: linux/amd64
          tags: ${{ needs.compute-tags.outputs.health_bridge_tags }}
          cache-from: type=gha,scope=ifix-ai-pod-health-bridge
          cache-to:   type=gha,scope=ifix-ai-pod-health-bridge,mode=max
          labels: |
            org.opencontainers.image.source=${{ github.server_url }}/${{ github.repository }}
            org.opencontainers.image.revision=${{ github.sha }}
            org.opencontainers.image.created=${{ github.event.head_commit.timestamp }}

  summary:
    name: Summary
    runs-on: ubuntu-latest
    needs: [build-pod, build-health-bridge, compute-tags]
    if: always()
    steps:
      - name: Print summary
        env:
          POD_TAGS: ${{ needs.compute-tags.outputs.pod_tags }}
          HB_TAGS: ${{ needs.compute-tags.outputs.health_bridge_tags }}
          STABLE: ${{ needs.compute-tags.outputs.is_stable_release }}
        run: |
          {
            echo "## Build Summary"
            echo ""
            echo "**Stable release:** ${STABLE}"
            echo ""
            echo "### Pod image tags"
            echo '```'
            echo "${POD_TAGS}"
            echo '```'
            echo ""
            echo "### Health-bridge image tags"
            echo '```'
            echo "${HB_TAGS}"
            echo '```'
            echo ""
            echo "### Next steps"
            echo "- Run smoke-test: \`gh workflow run smoke.yml -f image_tag=<sha-tag>\`"
            echo "- Stable promotion: gates D-19 green in smoke-report then \`git tag vX.Y.Z\` + push"
          } >> "$GITHUB_STEP_SUMMARY"
```

Key design decisions locked into this file:
- Four jobs: `test` (Go + template integrity), `compute-tags` (centralized tag math), `build-pod`, `build-health-bridge`. `build-*` depend on `test` + `compute-tags`.
- `test` runs BEFORE any push — a broken build cannot produce an image.
- Template SHA-256 check repeats the Dockerfile's build-time check in CI for earlier failure signal (saves 5-10 min vs waiting for the Docker layer step).
- Tag computation handles 4 scenarios: tag push (D-23 stable), main, develop, manual dispatch, PR fallback.
- Stable release (`refs/tags/v*`) adds `:latest` + `:vX.Y.Z` + `:vX.Y.Z-sha` simultaneously.
- Develop adds `:latest-dev` as a rolling convenience (matches converseai-v4 pattern).
- Labels apply OCI standard annotations (image.source, image.revision, image.created).
- `platforms: linux/amd64` only — Vast.ai has no arm64 GPUs per 01-CONTEXT.md `<deferred>` multi-arch.
- `cache-from`/`cache-to` use GHA cache with distinct scopes per image (pod vs health-bridge independent caches).
- Permissions minimal: contents:read + packages:write.
- Concurrency cancel-in-progress: true — abort superseded builds.
- `summary` job always runs → writes a markdown step summary with resolved tags + next-step hint.

- Tag computation is deterministic: given the same REF + SHA, the outputs are identical (shell logic only)

## Trust Boundaries

| Boundary | Description |
|---|---|
| GitHub Actions runner → GHCR | Image push authenticated via GITHUB_TOKEN (auto-issued, scoped to packages:write on this repo) |
| Forked PR → build-pod.yml | PR from external fork could attempt to inject malicious Dockerfile content |
| External base images (nvidia/cuda, golang, distroless, llama.cpp, speaches, infinity, dcgm) → our image | Supply chain from upstream registries |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|---|---|---|---|---|---|
| T-01-07-01 | Tampering (supply chain) | pod/Dockerfile pulls `ghcr.io/ggml-org/llama.cpp:server-cuda` etc. by tag | medium | mitigate | For Phase 1 smoke, `:server-cuda` / `:latest` used. Upgrade path: pin by digest in follow-up (noted in plan 03 threat model T-01-03-06). Documented as accepted risk for Phase 1 gated by smoke-test's D-19 gates — any compromised base would almost certainly fail `tool_call_valid` or produce OOM. |
| T-01-07-02 | Tampering | Forked PR Dockerfile injection | medium | mitigate | `paths:` filter limits auto-build triggers. Workflow does NOT build on PRs from external forks automatically — `permissions: packages: write` is scoped to push events on protected branches (main, develop) and tags. PR builds produce `:pr-{sha}` but require `workflow_dispatch` (manual) to push. |
| T-01-07-03 | Elevation of Privilege | GITHUB_TOKEN | low | accept | Auto-issued token scoped minimally to this repo's packages. Expires after job. Cannot modify code or protected branches. |
| T-01-07-04 | Tampering | Qwen template drift between plan 02 commit and build | high | mitigate | Dedicated `test` job step verifies `pod/templates/qwen3.5-27b-tool-calling.jinja` SHA-256 matches `.sha256` sidecar. | T-01-07-05 | Information Disclosure | Build logs | low | accept | No secrets leaked — GITHUB_TOKEN is scrubbed by GHA. Docker build args are public (no secret-mount). |
| T-01-07-06 | DoS | Runaway builds | low | mitigate | `timeout-minutes: 60` on build-pod, `15` on health-bridge, `10` on test. Concurrency cancel-in-progress prevents stacking on same branch. |

`high` severity T-01-07-04 has explicit mitigation (pre-build integrity check). `medium` severities have stated upgrade paths.
